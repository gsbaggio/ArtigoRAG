class Solution:
    def rearrangeArray(self, nums: list[int]) -> list[int]:
        n = len(nums)
        nums.sort()
        
        result = [0] * n
        # Place elements in alternating positions: smallest, largest, second smallest, second largest, etc.
        left, right = 0, n - 1
        for i in range(n):
            if i % 2 == 0:
                result[i] = nums[left]
                left += 1
            else:
                result[i] = nums[right]
                right -= 1
        
        return result
