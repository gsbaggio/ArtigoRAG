from typing import List

class Solution:
    def rearrangeArray(self, nums: List[int]) -> List[int]:
        nums.sort()
        result = [0] * len(nums)
        
        i, j = 0, len(nums)//2
        for k in range(len(nums)):
            if k % 2 == 0:
                result[k] = nums[i]
                i += 1
            else:
                result[k] = nums[j]
                j += 1
        
        return result
