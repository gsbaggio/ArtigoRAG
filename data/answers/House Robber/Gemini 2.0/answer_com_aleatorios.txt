class Solution {
public:
    int rob(vector<int>& nums) {
        int n = nums.size();
        if (n == 0) return 0;
        if (n == 1) return nums[0];

        int prev2 = nums[0]; // Represents dp[i-2]
        int prev1 = max(nums[0], nums[1]); // Represents dp[i-1]
        int current; // Represents dp[i]

        for (int i = 2; i < n; i++) {
            current = max(prev2 + nums[i], prev1);
            prev2 = prev1;
            prev1 = current;
        }

        return prev1;
    }
};
