#include <vector>
#include <algorithm>

class Solution {
public:
    int rob(std::vector<int>& nums) {
        int n = nums.size();
        if (n == 0) {
            return 0;
        } else if (n == 1) {
            return nums[0];
        }

        // Initialize dp array to store the maximum money that can be robbed up to that house
        std::vector<int> dp(n, 0);

        // Base cases
        dp[0] = nums[0];
        dp[1] = std::max(nums[0], nums[1]);

        // Fill the dp array iteratively
        for (int i = 2; i < n; ++i) {
            dp[i] = std::max(dp[i - 1], dp[i - 2] + nums[i]);
        }

        return dp[n - 1];
    }
};
