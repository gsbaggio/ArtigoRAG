class Solution:
    def minCapability(self, nums: List[int], k: int) -> int:
        def can_steal_k_houses(capability):
            """Returns True if robber can steal at least k houses with given capability."""
            # Count houses that can be robbed with current capability
            count = 0
            i = 0
            while i < len(nums):
                if nums[i] <= capability:
                    count += 1
                    i += 2  # Skip adjacent house
                else:
                    i += 1
                
                if count >= k:
                    return True
            
            return False
        
        # Binary search on the capability value
        left = min(nums)
        right = max(nums)
        
        while left < right:
            mid = (left + right) // 2
            if can_steal_k_houses(mid):
                right = mid
            else:
                left = mid + 1
                
        return left
