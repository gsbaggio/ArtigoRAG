#include <algorithm>
#include <vector>

class Solution {
public:
    TreeNode* reverseOddLevels(TreeNode* root) {
        reverseOddLevelsHelper(root->left, root->right, 1);
        return root;
    }

private:
    void reverseOddLevelsHelper(TreeNode* node1, TreeNode* node2, int level) {
        if (!node1) return;

        if (level % 2 != 0) {
            std::swap(node1->val, node2->val);
        }

        reverseOddLevelsHelper(node1->left, node2->right, level + 1);
        reverseOddLevelsHelper(node1->right, node2->left, level + 1);
    }
};
