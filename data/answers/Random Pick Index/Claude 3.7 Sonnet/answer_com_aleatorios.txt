class Solution {
    private int[] nums;
    private Random rand;
    
    public Solution(int[] nums) {
        this.nums = nums;
        this.rand = new Random();
    }
    
    public int pick(int target) {
        int count = 0;
        int result = -1;
        
        // Reservoir sampling algorithm
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == target) {
                count++;
                // With probability 1/count, replace the result with current index
                if (rand.nextInt(count) == 0) {
                    result = i;
                }
            }
        }
        
        return result;
    }
}
