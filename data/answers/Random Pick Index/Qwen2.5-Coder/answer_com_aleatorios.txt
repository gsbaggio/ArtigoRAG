#include <vector>
#include <cstdlib> // for rand()

class Solution {
private:
    std::vector<int> nums;

public:
    Solution(std::vector<int>& nums) {
        this->nums = nums;
    }
    
    int pick(int target) {
        int res = -1; // Initialize result index
        int count = 0; // Count occurrences of target
        
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] == target) {
                count++;
                // Generate a random number between 0 and count-1
                if (rand() % count == 0) {
                    res = i;
                }
            }
        }
        
        return res;
    }
};
