from typing import List

class Solution:
    def getFinalState(self, nums: List[int], k: int, multiplier: int) -> List[int]:
        arr = nums.copy()
        n = len(arr)
        for _ in range(k):
            min_val = arr[0]
            min_index = 0
            for i in range(1, n):
                if arr[i] < min_val:
                    min_val = arr[i]
                    min_index = i
            arr[min_index] = min_val * multiplier
        return arr
