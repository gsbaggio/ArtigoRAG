class Solution:
    def hIndex(self, citations):
        # Step 1: Sort the citations array in ascending order
        citations.sort()
        
        # Step 2: Initialize variables
        n = len(citations)
        h_index = 0
        
        # Step 3: Iterate through the sorted array
        for i in range(n):
            # Calculate the number of papers with citations >= citations[i]
            h = n - i
            
            # Check if the current citation count satisfies the h-index condition
            if citations[i] >= h:
                h_index = h
                break
        
        return h_index
