class Solution:
    def uniqueXorTriplets(self, nums):
        unique_xor_values = set()
        n = len(nums)
        
        # Iterate over all possible triplets (i, j, k) with i <= j <= k
        for i in range(n):
            for j in range(i, n):
                for k in range(j, n):
                    xor_triplet = nums[i] ^ nums[j] ^ nums[k]
                    unique_xor_values.add(xor_triplet)
        
        # Return the number of unique XOR triplet values
        return len(unique_xor_values)
