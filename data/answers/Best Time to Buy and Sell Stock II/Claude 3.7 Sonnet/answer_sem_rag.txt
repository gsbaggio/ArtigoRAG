class Solution:
    def maxProfit(self, prices: list[int]) -> int:
        total_profit = 0
        
        for i in range(1, len(prices)):
            # If today's price is higher than yesterday's, we can make a profit
            if prices[i] > prices[i-1]:
                total_profit += prices[i] - prices[i-1]
                
        return total_profit
