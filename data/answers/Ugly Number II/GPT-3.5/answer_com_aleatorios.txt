#include <vector>
#include <algorithm>

class Solution {
public:
    int nthUglyNumber(int n) {
        if (n <= 0) return 0;
        
        std::vector<int> ugly(n, 1);
        int p2 = 0, p3 = 0, p5 = 0; // Pointers for 2, 3, 5
        
        for (int i = 1; i < n; ++i) {
            ugly[i] = std::min({ugly[p2] * 2, ugly[p3] * 3, ugly[p5] * 5});
            
            if (ugly[i] == ugly[p2] * 2) p2++;
            if (ugly[i] == ugly[p3] * 3) p3++;
            if (ugly[i] == ugly[p5] * 5) p5++;
        }
        
        return ugly[n - 1];
    }
};
