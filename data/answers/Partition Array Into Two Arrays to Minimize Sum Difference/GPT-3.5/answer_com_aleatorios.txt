public class Solution {
    public int minimumDifference(int[] nums) {
        Arrays.sort(nums);
        
        int n = nums.length;
        int minDiff = Integer.MAX_VALUE;
        
        for (int i = 0; i < n / 2; i++) {
            minDiff = Math.min(minDiff, nums[2*i+1] - nums[2*i]);
        }
        
        return minDiff;
    }
}
