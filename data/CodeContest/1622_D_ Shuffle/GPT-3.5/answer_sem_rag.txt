MOD = 998244353

def factorial(n):
    result = 1
    for i in range(1, n + 1):
        result = (result * i) % MOD
    return result

def mod_inverse(n):
    return pow(n, MOD - 2, MOD)

def nCr(n, r):
    return (factorial(n) * mod_inverse(factorial(r)) * mod_inverse(factorial(n - r))) % MOD

def count_strings(n, k, s):
    ones = [i for i, c in enumerate(s) if c == '1']
    
    result = 0
    for i in range(len(ones) - k + 1):
        result = (result + nCr(len(ones) - i - 1, k - 1)) % MOD
    
    return result

# Input
n, k = map(int, input().split())
s = input().strip()

# Output
print(count_strings(n, k, s))
