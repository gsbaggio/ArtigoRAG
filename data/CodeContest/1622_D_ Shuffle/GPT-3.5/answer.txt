MOD = 998244353

def count_strings(n, k, s):
    dp = [[[0 for _ in range(k + 1)] for _ in range(n + 1)] for _ in range(n + 1)]
    cnt_ones = [0] * (n + 1)
    
    for i in range(1, n + 1):
        cnt_ones[i] = cnt_ones[i - 1] + (s[i - 1] == '1')
    
    dp[0][0][0] = 1
    
    for i in range(1, n + 1):
        for j in range(0, cnt_ones[i] + 1):
            for l in range(0, k + 1):
                for m in range(max(0, cnt_ones[i - 1] - j), min(j, l) + 1):
                    dp[i][j][l] += dp[i - 1][cnt_ones[i - 1]][l - m]
                    dp[i][j][l] %= MOD
    
    result = sum(dp[n][j][k] for j in range(cnt_ones[n] + 1)) % MOD
    
    return result

# Reading input
n, k = map(int, input().split())
s = input().strip()

# Calculating and printing the result
result = count_strings(n, k, s)
print(result)
