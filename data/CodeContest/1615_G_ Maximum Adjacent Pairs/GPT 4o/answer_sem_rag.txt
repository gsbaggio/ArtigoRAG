def maximize_array_value(n, a):
    # Result array
    result = a[:]
    
    # Assign values to zeros
    for i in range(n):
        if result[i] == 0:
            # Find a value to replace the zero
            left = result[i - 1] if i > 0 else 0
            right = result[i + 1] if i < n - 1 else 0
            
            # Choose a value that maximizes adjacent pairs
            for value in range(1, n + 1):
                if value != left and value != right:
                    result[i] = value
                    break
    
    return result

# Input reading
n = int(input())
a = list(map(int, input().split()))

# Solve and print the result
result = maximize_array_value(n, a)
print(" ".join(map(str, result)))
