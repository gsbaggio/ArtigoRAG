def maximize_array_value(n, a):
    # Result array
    result = a[:]
    
    for i in range(n):
        if a[i] == 0:
            # Get the neighbors
            left = result[i - 1] if i > 0 else None
            right = result[i + 1] if i < n - 1 else None
            
            # Try to pick a number that maximizes distinct adjacent pairs
            for num in range(1, n + 1):
                if num != left and num != right:
                    result[i] = num
                    break
    
    return result

# Input reading
if __name__ == "__main__":
    import sys
    input = sys.stdin.read
    data = input().split()
    
    n = int(data[0])
    a = list(map(int, data[1:]))
    
    # Solve the problem
    result = maximize_array_value(n, a)
    
    # Output the result
    print(" ".join(map(str, result)))
