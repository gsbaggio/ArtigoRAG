def maximize_array_value(n, a):
    # Result array
    result = a[:]
    
    # Iterate through the array
    for i in range(n):
        if result[i] == 0:
            # Set of possible replacements
            available = set(range(1, n + 1))
            
            # Remove adjacent values from the available set
            if i > 0:
                available.discard(result[i - 1])
            if i < n - 1:
                available.discard(result[i + 1])
            
            # Replace zero with any available number
            result[i] = available.pop()
    
    return result

# Input reading
if __name__ == "__main__":
    import sys
    input = sys.stdin.read
    data = input().split()
    
    n = int(data[0])
    a = list(map(int, data[1:]))
    
    # Solve the problem
    result = maximize_array_value(n, a)
    
    # Print the result
    print(" ".join(map(str, result)))
