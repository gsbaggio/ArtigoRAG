#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    vector<bool> used(n + 1, false);
    for (int i = 0; i < n; i++) {
        if (a[i] != 0) {
            used[a[i]] = true;
        }
    }

    // First pass: Extend existing pairs
    for (int i = 0; i < n - 1; i++) {
        if (a[i] == 0 && a[i + 1] != 0) {
            a[i] = a[i + 1];
        } else if (a[i] != 0 && a[i + 1] == 0) {
            a[i + 1] = a[i];
        }
    }

    // Second pass: Fill remaining zeros
    int next_unused = 1;
    for (int i = 0; i < n; i++) {
        if (a[i] == 0) {
            while (next_unused <= n && used[next_unused]) {
                next_unused++;
            }
            a[i] = next_unused;
            used[next_unused] = true;
        }
    }

    // Output the modified array
    for (int i = 0; i < n; i++) {
        cout << a[i] << " ";
    }
    cout << "\n";

    return 0;
}
