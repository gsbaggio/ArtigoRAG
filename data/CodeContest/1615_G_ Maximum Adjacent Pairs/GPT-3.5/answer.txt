def maximize_array_value(n, arr):
    max_val = max(arr.count(0), 1)
    current = 1

    for i in range(n):
        if arr[i] == 0:
            if max_val == 1:
                arr[i] = current
            else:
                arr[i] = current % max_val + 1
                current += 1

    for i in range(n):
        if arr[i] == max_val:
            if i == 0:
                arr[i] = max_val - 1
            elif i == n - 1:
                arr[i] = max_val - 1
            elif arr[i - 1] == max_val or arr[i + 1] == max_val:
                arr[i] = max_val - 1
            else:
                arr[i] = max_val

    return arr

# Input
n = int(input())
arr = list(map(int, input().split()))

# Output
result = maximize_array_value(n, arr)
print(*result)
