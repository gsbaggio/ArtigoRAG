def max_possible_array(n, a):
    zeros_count = a.count(0)
    unique_zeros = list(range(1, zeros_count + 1))
    non_zeros = [i for i in range(1, n + 1) if i not in unique_zeros]

    j = 0
    for i in range(n):
        if a[i] == 0:
            a[i] = unique_zeros[j]
            j += 1

    prev = -1
    for i in range(n):
        if a[i] == prev:
            for val in non_zeros:
                if val != prev:
                    a[i] = val
                    prev = val
                    break
        else:
            prev = a[i]

    return a

# Input
n = int(input())
a = list(map(int, input().split()))

# Output
result = max_possible_array(n, a)
print(*result)
