def solve():
    q = int(input())
    arr = []
    
    for _ in range(q):
        query = list(map(int, input().split()))
        
        if query[0] == 1:
            arr.append(query[1])
        elif query[0] == 2:
            x = query[1]
            y = query[2]
            for i in range(len(arr)):
                if arr[i] == x:
                    arr[i] = y
    
    print(*arr)

solve()
