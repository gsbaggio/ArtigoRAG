#include <bits/stdc++.h>
using namespace std;

bool is_sorted(vector<int>& a) {
    for (size_t i = 1; i < a.size(); ++i) {
        if (a[i] < a[i - 1]) {
            return false;
        }
    }
    return true;
}

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector<int> a(n);
        for (int i = 0; i < n; ++i) {
            cin >> a[i];
        }
        
        int k = 0;
        while (!is_sorted(a)) {
            int x = a[n - 1];
            vector<int> left, right;
            for (int i = 0; i < n - 1; ++i) {
                if (a[i] <= x) {
                    left.push_back(a[i]);
                } else {
                    right.push_back(a[i]);
                }
            }
            left.push_back(x);
            a = left;
            a.insert(a.end(), right.begin(), right.end());
            ++k;
        }
        
        cout << k << "\n";
    }
    return 0;
}
