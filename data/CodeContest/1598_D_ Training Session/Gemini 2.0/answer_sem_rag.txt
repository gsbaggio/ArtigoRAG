#include <iostream>
#include <vector>

using namespace std;

long long solve() {
    int n;
    cin >> n;

    vector<pair<int, int>> problems(n);
    for (int i = 0; i < n; ++i) {
        cin >> problems[i].first >> problems[i].second;
    }

    long long count = 0;
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                // Check if the topics are different
                bool topics_different = (problems[i].first != problems[j].first) &&
                                        (problems[i].first != problems[k].first) &&
                                        (problems[j].first != problems[k].first);

                // Check if the difficulties are different
                bool difficulties_different = (problems[i].second != problems[j].second) &&
                                              (problems[i].second != problems[k].second) &&
                                              (problems[j].second != problems[k].second);

                if (topics_different || difficulties_different) {
                    count++;
                }
            }
        }
    }

    return count;
}

int main() {
    int t;
    cin >> t;

    while (t--) {
        cout << solve() << endl;
    }

    return 0;
}
