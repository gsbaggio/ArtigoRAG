def minimize_score(t, test_cases):
    results = []
    for case in test_cases:
        n, k, arr = case
        arr.sort()  # Sort the array in ascending order
        
        # Perform k operations
        score = 0
        for i in range(k):
            score += arr[i] // arr[n - k + i]
        
        # Add the remaining elements
        score += sum(arr[k:n - k])
        results.append(score)
    
    return results


# Input Reading
t = int(input())
test_cases = []
for _ in range(t):
    n, k = map(int, input().split())
    arr = list(map(int, input().split()))
    test_cases.append((n, k, arr))

# Solve and Output Results
results = minimize_score(t, test_cases)
for res in results:
    print(res)
