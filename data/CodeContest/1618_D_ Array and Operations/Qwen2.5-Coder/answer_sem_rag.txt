def min_score(t, test_cases):
    results = []
    
    for n, k, arr in test_cases:
        # Sort the array
        arr.sort()
        
        score = 0
        i, j = 0, n - 1
        
        # Perform k operations
        for _ in range(k):
            score += arr[i] // arr[j]
            i += 1
            j -= 1
        
        # Add the remaining elements to the score
        score += sum(arr[i:j+1])
        
        results.append(score)
    
    return results

# Read input
t = int(input())
test_cases = []

for _ in range(t):
    n, k = map(int, input().split())
    arr = list(map(int, input().split()))
    test_cases.append((n, k, arr))

# Get results
results = min_score(t, test_cases)

# Print results
for result in results:
    print(result)
