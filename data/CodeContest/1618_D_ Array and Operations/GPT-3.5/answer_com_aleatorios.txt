import math

def min_score(t, test_cases):
    results = []
    
    for case in test_cases:
        n, k = case[0]
        a = case[1]
        
        a.sort(reverse=True)
        
        score = 0
        
        for i in range(k):
            score += a.pop(0) // a.pop()
        
        score += sum(a)
        
        results.append(score)
    
    return results

# Input parsing
t = int(input())
test_cases = []

for _ in range(t):
    n, k = map(int, input().split())
    a = list(map(int, input().split()))
    test_cases.append((n, k, a))

# Calculate and output the results
results = min_score(t, test_cases)
for res in results:
    print(res)
