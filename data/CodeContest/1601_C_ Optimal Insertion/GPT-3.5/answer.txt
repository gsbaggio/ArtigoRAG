from bisect import bisect_right

def count_inversions(a, b):
    a.sort()
    inversions = 0
    for num in b:
        inversions += bisect_right(a, num)
    return inversions

def min_inversions(t, test_cases):
    results = []
    for _ in range(t):
        n, m, a, b = test_cases[_]
        result = count_inversions(a, b)
        results.append(result)
    return results

# Input
t = 3
test_cases = [
    (3, 4, [1, 2, 3], [4, 3, 2, 1]),
    (3, 3, [3, 2, 1], [1, 2, 3]),
    (5, 4, [1, 3, 5, 3, 1], [4, 3, 6, 1])
]

# Output
results = min_inversions(t, test_cases)
for result in results:
    print(result)
