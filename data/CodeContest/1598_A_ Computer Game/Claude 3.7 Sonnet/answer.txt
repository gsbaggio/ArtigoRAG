#include <iostream>
#include <vector>
#include <queue>
using namespace std;

bool canCompleteLevel(const vector<string>& grid, int n) {
    // Grid is 2Ã—n
    vector<vector<bool>> visited(2, vector<bool>(n, false));
    
    // Define 8 possible movements: horizontal, vertical, and diagonal
    const int dx[] = {-1, -1, -1, 0, 0, 1, 1, 1};
    const int dy[] = {-1, 0, 1, -1, 1, -1, 0, 1};
    
    queue<pair<int, int>> q;
    q.push({0, 0});  // 0-indexed, so (0,0) is the top-left cell
    visited[0][0] = true;
    
    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();
        
        // If we reached the destination (bottom-right)
        if (x == 1 && y == n - 1) {
            return true;
        }
        
        // Try all 8 directions
        for (int i = 0; i < 8; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            
            // Check if the new position is valid
            if (nx >= 0 && nx < 2 && ny >= 0 && ny < n && 
                grid[nx][ny] == '0' && !visited[nx][ny]) {
                visited[nx][ny] = true;
                q.push({nx, ny});
            }
        }
    }
    
    // No path found
    return false;
}

int main() {
    int t;
    cin >> t;
    
    while (t--) {
        int n;
        cin >> n;
        
        vector<string> grid(2);
        cin >> grid[0] >> grid[1];
        
        if (canCompleteLevel(grid, n)) {
            cout << "YES" << endl;
        } else {
            cout << "NO" << endl;
        }
    }
    
    return 0;
}
