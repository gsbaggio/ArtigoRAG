t = int(input())  # Number of test cases

for _ in range(t):
    n = int(input())  # Number of columns
    grid = [input() for _ in range(2)]  # Input grid

    # Starting position
    x, y = 0, 0
    possible = True

    while y < n - 1:
        if grid[x][y] == '1':  # If trap cell encountered
            possible = False
            break

        if x == 0:
            x = 1
        else:
            x = 0
            y += 1

    if possible:
        print("YES")
    else:
        print("NO")
