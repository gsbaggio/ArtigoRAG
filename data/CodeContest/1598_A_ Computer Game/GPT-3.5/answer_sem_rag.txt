def is_completable_level(n, level):
    reachable = [[False for _ in range(n)] for _ in range(2)]
    reachable[0][0] = True  # Starting cell is reachable

    for col in range(1, n):
        for row in range(2):
            if level[row][col] == '0':
                for k in range(3):
                    if 0 <= row - k < 2 and reachable[row - k][col - 1]:
                        reachable[row][col] = True
                        break

    return "YES" if reachable[1][n - 1] else "NO"

# Read input
t = int(input())
for _ in range(t):
    n = int(input())
    level = [input() for _ in range(2)]
    print(is_completable_level(n, level))
