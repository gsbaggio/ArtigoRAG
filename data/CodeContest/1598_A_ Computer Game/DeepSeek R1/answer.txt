t = int(input().strip())
results = []
for _ in range(t):
    n = int(input().strip())
    row0 = input().strip()
    row1 = input().strip()
    found = False
    for i in range(1, n-1):
        if row0[i] == '1' and row1[i] == '1':
            found = True
            break
    results.append('NO' if found else 'YES')

for res in results:
    print(res)
