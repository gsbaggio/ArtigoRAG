t = int(input().strip())
for _ in range(t):
    n = int(input().strip())
    row1 = input().strip()
    row2 = input().strip()
    possible = True
    for i in range(1, n - 1):
        if row1[i] == '1' and row2[i] == '1':
            possible = False
            break
    print("YES" if possible else "NO")
