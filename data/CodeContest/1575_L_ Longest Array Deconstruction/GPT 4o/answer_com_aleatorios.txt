def maximize_f_a(n, a):
    # Initialize the counter for f(a)
    count = 0
    
    # Iterate through the array
    for i in range(1, n + 1):  # 1-based indexing
        if a[i - 1] == i:  # Check if a[i] == i
            count += 1
    
    # Return the maximum f(a)
    return count

# Input reading
if __name__ == "__main__":
    import sys
    input = sys.stdin.read
    data = input().split()
    
    n = int(data[0])  # Length of the sequence
    a = list(map(int, data[1:]))  # Sequence
    
    # Output the result
    print(maximize_f_a(n, a))
