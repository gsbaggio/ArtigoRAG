def max_f_a(n, a):
    max_f = 0
    
    if a[0] == 1:
        max_f += 1
    
    if a[-1] == n:
        max_f += 1
    
    f = 0
    for i in range(1, n):
        if a[i] == i:
            f += 1
    
    max_f = max(max_f, f)
    
    f = 0
    for i in range(n - 2, -1, -1):
        if a[i] == n - i:
            f += 1
    
    return max(max_f, f)

# Input
n = int(input())
a = list(map(int, input().split()))

# Output
print(max_f_a(n, a))
