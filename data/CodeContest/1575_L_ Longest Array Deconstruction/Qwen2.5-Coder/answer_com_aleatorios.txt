#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    int count = 0;
    for (int i = 0; i < n; ++i) {
        if (a[i] == i + 1) { // Note: i is 0-based, so we check a[i] == i + 1
            ++count;
        }
    }
    
    cout << count;
    
    return 0;
}
