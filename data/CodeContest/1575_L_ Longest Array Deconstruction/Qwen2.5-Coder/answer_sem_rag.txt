def max_f_a(n, a):
    # Step 1: Count occurrences of each number in the sequence
    count = {}
    for num in a:
        if num in count:
            count[num] += 1
        else:
            count[num] = 1
    
    # Step 2: Determine the maximum f(a)
    max_f = 0
    for i in range(1, n + 1):
        if i in count:
            # We can place at most one occurrence of 'i' at index 'i'
            max_f += min(count[i], 1)
    
    return max_f

# Read input
import sys
input = sys.stdin.read().split()
n = int(input[0])
a = list(map(int, input[1:]))

# Output the result
print(max_f_a(n, a))
