def solve():
    n, k, x = map(int, input().split())
    s = input()
    
    groups = []
    count = 0
    for char in s:
        if char == '*':
            count += 1
        else:
            if count > 0:
                groups.append(count)
                count = 0
            groups.append(-1)  # -1 represents 'a'
    if count > 0:
        groups.append(count)
    
    
    nums = []
    for g in groups:
        if g == -1:
            nums.append(-1)
        else:
            nums.append(g * k + 1)
            
    x -= 1
    
    result = ""
    for num in nums:
        if num == -1:
            result += 'a'
        else:
            b_count = x % num
            result += 'b' * b_count
            x //= num
            
    print(result)

t = int(input())
for _ in range(t):
    solve()
