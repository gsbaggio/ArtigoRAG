def solve():
    n, k, x = map(int, input().split())
    s = input()
    
    groups = []
    count = 0
    for char in s:
        if char == '*':
            count += 1
        else:
            if count > 0:
                groups.append(count * k + 1)
                count = 0
            groups.append(-1)  # Mark 'a' with -1
    if count > 0:
        groups.append(count * k + 1)
    
    x -= 1
    result = ""
    for i in range(len(groups) - 1, -1, -1):
        if groups[i] == -1:
            result = "a" + result
        else:
            num_bs = x % groups[i]
            result = "b" * num_bs + result
            x //= groups[i]
            
    print(result)

t = int(input())
for _ in range(t):
    solve()
