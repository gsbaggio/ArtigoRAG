def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    t = int(data[0])  # Number of test cases
    idx = 1
    
    results = []
    
    for _ in range(t):
        n, k, x = map(int, data[idx:idx+3])
        s = data[idx+3]
        idx += 4
        
        x -= 1  # Convert x to 0-based index
        result = []
        segments = []
        
        # Parse the string into segments of 'a' and '*'
        i = 0
        while i < n:
            if s[i] == 'a':
                result.append('a')
            else:
                # Count consecutive '*'s
                count = 0
                while i < n and s[i] == '*':
                    count += 1
                    i += 1
                i -= 1
                segments.append(count)
            i += 1
        
        # Process each segment of '*'
        for count in segments:
            # Determine how many 'b's to place for this segment
            num_b = x % (k * count + 1)
            result.append('b' * num_b)
            x //= (k * count + 1)
        
        results.append(''.join(result))
    
    sys.stdout.write('\n'.join(results) + '\n')
