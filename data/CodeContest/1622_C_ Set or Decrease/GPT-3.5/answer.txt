def min_steps_to_reduce_sum(t, test_cases):
    for n, k, a in test_cases:
        total_sum = sum(a)
        if total_sum <= k:
            print(0)
        else:
            a.sort(reverse=True)
            steps = 0
            for i in range(n):
                if total_sum <= k:
                    break
                diff = total_sum - k
                steps += min(a[i], diff)
                total_sum -= a[i]
            print(steps)

# Test Input
t = 4
test_cases = [
    (1, 10, [20]),
    (2, 69, [6, 9]),
    (7, 8, [1, 2, 1, 3, 1, 2, 1]),
    (10, 1, [1, 2, 3, 1, 2, 6, 1, 6, 8, 10])
]

min_steps_to_reduce_sum(t, test_cases)
