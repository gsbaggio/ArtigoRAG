def solve():
    t = int(input())
    for _ in range(t):
        n, k = map(int, input().split())
        a = list(map(int, input().split()))
        
        total = sum(a)
        if total <= k:
            print(0)
            continue
        
        # Need to reduce the sum by this amount
        need_to_reduce = total - k
        
        # Direct decrease: one step per unit reduction
        ans = need_to_reduce
        
        # Try setting all elements equal to a[i], then decreasing them if needed
        for i in range(n):
            # Make all elements equal to a[i] or smaller
            ops = 0
            new_sum = 0
            
            for j in range(n):
                if a[j] <= a[i]:
                    new_sum += a[j]  # Keep original values for smaller elements
                else:
                    ops += 1  # Operation to set a[j] = a[i]
                    new_sum += a[i]  # New value after setting equal
            
            # If sum is still too large, we need to decrease
            if new_sum > k:
                ops += new_sum - k  # One operation per unit to decrease
            
            ans = min(ans, ops)
        
        print(ans)

solve()
