#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int t;
    cin >> t;
    
    while(t--){
        int n;
        cin >> n;
        
        vector<long long> prices(n);
        for(auto &price : prices){
            cin >> price;
        }
        
        int max_coins = 0;
        
        for(const auto &price : prices){
            int c3 = price / 3;
            int remaining = price % 3;
            
            int c2 = 0, c1 = 0;
            if(remaining == 1){
                c1 = 1;
            } else if(remaining == 2){
                c2 = 1;
            }
            
            int total_coins = c3 + c2 + c1;
            max_coins = max(max_coins, total_coins);
        }
        
        cout << max_coins << "\n";
    }
    
    return 0;
}
