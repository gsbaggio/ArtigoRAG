#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

long long min_coins(long long price) {
    long long quotient = price / 3;
    long long remainder = price % 3;
    
    if (remainder == 0) {
        return quotient;
    } else if (remainder == 1) {
        if (quotient == 0) {
            return 1;
        } else {
            return quotient - 1 + 2; // (q-1) 3-burle coins + two 2-burle coins
        }
    } else { // remainder == 2
        return quotient + 1; // q 3-burle coins + one 2-burle coin
    }
}

int main() {
    int t;
    cin >> t;
    
    while (t--) {
        int n;
        cin >> n;
        
        long long max_coins = 0;
        for (int i = 0; i < n; i++) {
            long long price;
            cin >> price;
            max_coins = max(max_coins, min_coins(price));
        }
        
        cout << max_coins << endl;
    }
    
    return 0;
}
