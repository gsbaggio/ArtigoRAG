def solve():
    l, r = map(int, input().split())
    
    if l == r:
        print(0)
        return
    
    if (l & r) != 0:
        print(0)
        return
    
    ans = float('inf')
    
    for i in range(18):
        count = 0
        for j in range(l, r + 1):
            if (j >> i) & 1 == 0:
                count += 1
        ans = min(ans, count)
    
    print(ans)

t = int(input())
for _ in range(t):
    solve()
