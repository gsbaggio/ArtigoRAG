import sys

def main():
    data = sys.stdin.read().split()
    t = int(data[0])
    index = 1
    results = []
    for _ in range(t):
        n = int(data[index]); index += 1
        a = list(map(int, data[index:index+n]))
        index += n
        
        # Check if the array is sorted
        is_sorted = True
        for i in range(1, n):
            if a[i] < a[i-1]:
                is_sorted = False
                break
                
        if is_sorted:
            results.append("YES")
            continue
            
        # Check for duplicates
        seen = set()
        has_dup = False
        for num in a:
            if num in seen:
                has_dup = True
                break
            seen.add(num)
            
        if has_dup:
            results.append("YES")
            continue
            
        # Build the permutation d where d[i] = a[i] - 1
        d = [0] * n
        for i in range(n):
            d[i] = a[i] - 1
            
        visited = [False] * n
        cycles = 0
        for i in range(n):
            if not visited[i]:
                cycles += 1
                cur = i
                while not visited[cur]:
                    visited[cur] = True
                    cur = d[cur]
                    
        parity = (n - cycles) % 2
        if parity == 0:
            results.append("YES")
        else:
            results.append("NO")
            
    print("\n".join(results))

if __name__ == "__main__":
    main()
