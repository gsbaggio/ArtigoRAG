def solve():
    n = int(input())
    a = list(map(int, input().split()))

    if len(set(a)) != n:
        print("YES")
        return

    inversions = 0
    for i in range(n):
        for j in range(i + 1, n):
            if a[i] > a[j]:
                inversions += 1

    if inversions % 2 == 0:
        print("YES")
    else:
        print("NO")

t = int(input())
for _ in range(t):
    solve()
