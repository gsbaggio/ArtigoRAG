def solve():
    n = int(input())
    a = list(map(int, input().split()))

    def is_sortable(arr):
        inversions = 0
        for i in range(n):
            for j in range(i + 1, n):
                if arr[i] > arr[j]:
                    inversions += 1
        return inversions % 2 == 0

    if is_sortable(a):
        print("YES")
    else:
        print("NO")

t = int(input())
for _ in range(t):
    solve()
