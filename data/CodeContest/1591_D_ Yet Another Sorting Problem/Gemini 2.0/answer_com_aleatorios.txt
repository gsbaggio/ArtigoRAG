def solve():
    n = int(input())
    a = list(map(int, input().split()))

    if len(set(a)) == 1:
        print("YES")
        return

    inversions = 0
    for i in range(n):
        for j in range(i + 1, n):
            if a[i] > a[j]:
                inversions += 1

    if inversions % 2 == 0:
        print("YES")
    else:
        counts = {}
        for x in a:
            counts[x] = counts.get(x, 0) + 1
        
        has_duplicate = False
        for x in counts:
            if counts[x] > 1:
                has_duplicate = True
                break
        
        if has_duplicate:
            print("YES")
        else:
            print("NO")

t = int(input())
for _ in range(t):
    solve()
