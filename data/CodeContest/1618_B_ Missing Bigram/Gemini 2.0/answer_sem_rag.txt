def solve():
    n = int(input())
    bigrams = input().split()
    
    if n == 3:
        print(bigrams[0][0] + bigrams[0])
        return

    ans = ""
    ans += bigrams[0][0]
    
    for i in range(n - 2):
        ans += bigrams[i][1]
    
    if len(ans) == n - 1:
        ans += bigrams[-1][1]
    
    if len(ans) < n:
        ans += bigrams[-1][1]
    
    if len(ans) > n:
        ans = ans[:n]
    
    print(ans)

t = int(input())
for _ in range(t):
    solve()
