def solve():
    n = int(input())
    bigrams = input().split()
    
    ans = ""
    ans += bigrams[0]
    
    for i in range(1, len(bigrams)):
        if bigrams[i][0] != ans[-1]:
            ans += bigrams[i]
        else:
            ans += bigrams[i][1]
            
    if len(ans) < n:
        ans += 'a'
        
    print(ans)

t = int(input())
for _ in range(t):
    solve()
