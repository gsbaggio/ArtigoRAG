def solve_missing_bigram():
    t = int(input())
    
    for _ in range(t):
        n = int(input())  # Length of the original word
        bigrams = input().split()
        
        # Start the word with the first bigram
        word = bigrams[0]
        
        # Add subsequent letters by checking if bigrams connect
        for i in range(1, len(bigrams)):
            # If the last character of the word matches the first character of the bigram,
            # we only need to add the second character
            if word[-1] == bigrams[i][0]:
                word += bigrams[i][1]
            else:
                # This means there's a missing bigram between these two
                # Add the entire bigram
                word += bigrams[i]
        
        # If our word is too short (can happen if the last bigram was removed)
        # we add an arbitrary character (using 'a' here)
        if len(word) < n:
            word += 'a'
            
        print(word)

solve_missing_bigram()
