def solve():
    import sys
    input = sys.stdin.read
    data = input().splitlines()
    
    t = int(data[0])  # Number of test cases
    results = []
    idx = 1
    
    for _ in range(t):
        n = int(data[idx])  # Number of students
        idx += 1
        preferences = [list(map(int, data[idx + i].split())) for i in range(n)]
        idx += n
        
        possible = False
        
        # Check all pairs of days (day1, day2)
        for day1 in range(5):
            for day2 in range(day1 + 1, 5):
                only_day1 = 0
                only_day2 = 0
                both_days = 0
                
                for student in preferences:
                    if student[day1] and not student[day2]:
                        only_day1 += 1
                    elif student[day2] and not student[day1]:
                        only_day2 += 1
                    elif student[day1] and student[day2]:
                        both_days += 1
                
                # Check if we can divide into two groups of size n/2
                if only_day1 <= n // 2 and only_day2 <= n // 2:
                    remaining = n // 2 - only_day1 + n // 2 - only_day2
                    if remaining <= both_days:
                        possible = True
                        break
            if possible:
                break
        
        results.append("YES" if possible else "NO")
    
    sys.stdout.write("\n".join(results) + "\n")
