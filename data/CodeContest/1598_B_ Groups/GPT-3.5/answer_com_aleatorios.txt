def can_schedule_lessons(t, testcases):
    def check_schedule(n, students):
        days = [[0, 1, 2, 3, 4] for _ in range(n)]
        for i in range(n):
            for j in range(5):
                if students[i][j] == 0:
                    days[i].remove(j)

        for i in range(n):
            for j in range(i + 1, n):
                common_days = set(days[i]) & set(days[j])
                if common_days:
                    for day in common_days:
                        days[i].remove(day)
                        days[j].remove(day)

        for i in range(n):
            if len(days[i]) < 2:
                return 'NO'

        return 'YES'

    results = []
    for _ in range(t):
        n = testcases[_][0]
        students = testcases[_][1:]
        results.append(check_schedule(n, students))

    return results

# Input reading and function call
t = int(input())
testcases = []
for _ in range(t):
    n = int(input())
    students = [list(map(int, input().split())) for _ in range(n)]
    testcases.append((n, students))

results = can_schedule_lessons(t, testcases)
for res in results:
    print(res)
