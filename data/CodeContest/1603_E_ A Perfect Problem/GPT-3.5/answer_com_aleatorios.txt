def is_good(arr):
    return max(arr) * min(arr) >= sum(arr)

def count_perfect_sequences(n, M):
    def helper(arr, idx):
        nonlocal count
        if idx == n:
            if is_good(arr):
                count += 1
        else:
            for i in range(1, n+2):
                arr[idx] = i
                helper(arr, idx + 1)

    count = 0
    helper([0] * n, 0)
    return count % M

# Input parsing
n, M = map(int, input().split())
result = count_perfect_sequences(n, M)
print(result)
