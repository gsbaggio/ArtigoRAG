#include <iostream>
using namespace std;

long long modpow(long long base, long long exp, long long mod) {
    long long result = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) result = (result * base) % mod;
        base = (base * base) % mod;
        exp >>= 1;
    }
    return result;
}

int main() {
    int n;
    long long M;
    cin >> n >> M;
    
    // Result is n^n mod M
    cout << modpow(n, n, M) << endl;
    
    return 0;
}
