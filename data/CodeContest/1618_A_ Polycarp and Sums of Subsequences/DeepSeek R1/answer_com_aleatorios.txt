import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int t = scanner.nextInt();
        for (int i = 0; i < t; i++) {
            long[] b = new long[7];
            for (int j = 0; j < 7; j++) {
                b[j] = scanner.nextLong();
            }
            System.out.println(b[0] + " " + b[1] + " " + (b[6] - b[0] - b[1]));
        }
    }
}
