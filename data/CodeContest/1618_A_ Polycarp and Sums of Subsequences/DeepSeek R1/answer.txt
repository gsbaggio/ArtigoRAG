#include <iostream>
using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        long long b[7];
        for (int i = 0; i < 7; i++) {
            cin >> b[i];
        }
        long long x = b[0], y = b[1];
        long long z1 = b[2];
        if (b[3] == x + y && b[4] == x + z1 && b[5] == y + z1 && b[6] == x + y + z1) {
            cout << x << " " << y << " " << z1 << endl;
        } else {
            cout << x << " " << y << " " << b[3] << endl;
        }
    }
    return 0;
}
