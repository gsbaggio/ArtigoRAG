#include <iostream>
#include <vector>

using namespace std;

int main() {
  int t;
  cin >> t;

  while (t--) {
    int n;
    cin >> n;

    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
      cin >> a[i];
    }

    int height = 1;
    bool dead = false;

    for (int i = 0; i < n; ++i) {
      if (a[i] == 1) {
        if (i > 0 && a[i - 1] == 1) {
          height += 5;
        } else {
          height += 1;
        }
      } else {
        if (i > 0 && a[i - 1] == 0) {
          dead = true;
          break;
        }
      }
    }

    if (dead) {
      cout << -1 << endl;
    } else {
      cout << height << endl;
    }
  }

  return 0;
}
