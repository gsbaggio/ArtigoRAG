def flower_growth(t, test_cases):
    results = []
    for _ in range(t):
        n = int(input())
        a = list(map(int, input().split()))
        
        height = 1  # Initial height of the flower
        dry_days = 0  # Number of consecutive dry days
        
        for i in range(n):
            if a[i] == 1:
                dry_days = 0  # Reset dry days counter
                if i > 0 and a[i-1] == 1:
                    height += 5  # Grow by 5 cm if watered on both i and i-1
                else:
                    height += 1  # Grow by 1 cm if watered only on i
            else:
                dry_days += 1  # Increment dry days counter
                if dry_days == 2:
                    results.append(-1)  # Flower dies
                    break
        else:
            results.append(height)  # Append final height if flower didn't die
            
    return results

# Read input
t = int(input())
test_cases = [input().split() for _ in range(t)]

# Get results
results = flower_growth(t, test_cases)

# Print results
for result in results:
    print(result)
