#include <iostream>
#include <vector>

using namespace std;

int main() {
    int t;
    cin >> t;
    
    while (t--) {
        int n;
        cin >> n;
        
        vector<int> a(n);
        for (int i = 0; i < n; ++i) {
            cin >> a[i];
        }
        
        int height = 1; // Initial height of the flower
        bool died = false; // Flag to check if the flower died
        
        for (int i = 0; i < n; ++i) {
            if (a[i] == 0) { // Flower not watered today
                if (i > 0 && a[i-1] == 0) { // Flower not watered yesterday
                    died = true;
                    break;
                }
            } else { // Flower watered today
                if (i > 0 && a[i-1] == 1) { // Flower also watered yesterday
                    height += 5;
                } else {
                    height += 1;
                }
            }
        }
        
        if (died) {
            cout << "-1" << endl;
        } else {
            cout << height << endl;
        }
    }
    
    return 0;
}
