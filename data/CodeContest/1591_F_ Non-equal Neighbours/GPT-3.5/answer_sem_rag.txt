MOD = 998244353

def count_arrays(n, a):
    dp = [0] * (n + 1)
    dp[1] = a[0]
    
    for i in range(2, n + 1):
        dp[i] = (dp[i - 1] * min(a[i - 1], i - 1)) % MOD
    
    return dp[n]

# Reading input
n = int(input())
a = list(map(int, input().split()))

# Printing output
print(count_arrays(n, a))
