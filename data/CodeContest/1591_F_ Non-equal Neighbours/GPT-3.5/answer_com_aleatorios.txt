#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int MOD = 998244353;

int main() {
    int n;
    cin >> n;
    
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    long long result = 1;
    int prev = 0;

    for (int i = 0; i < n; ++i) {
        int curr = min(a[i], i + 1);
        result = (result * max(0, curr - prev)) % MOD;
        prev = max(prev, curr);
    }

    cout << result << endl;

    return 0;
}
