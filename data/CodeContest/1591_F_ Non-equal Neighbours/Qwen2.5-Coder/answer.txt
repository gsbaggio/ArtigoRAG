MOD = 998244353

def solve():
    n = int(input())
    a = list(map(int, input().split()))
    
    # Base case
    dp_prev = a[0]
    
    # Iterate through the array to compute dp values
    for i in range(1, n):
        dp_curr = (dp_prev * (a[i] - 1)) % MOD
        dp_prev = dp_curr
    
    # The final answer is dp[n]
    print(dp_curr)

if __name__ == "__main__":
    solve()
