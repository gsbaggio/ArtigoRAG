def solve():
    t = int(input())
    for _ in range(t):
        n, m = map(int, input().split())
        s = input()
        
        best_r, best_c = 1, 1
        max_moves = 0
        
        for start_r in range(1, n + 1):
            for start_c in range(1, m + 1):
                
                curr_r, curr_c = start_r, start_c
                moves = 0
                
                for move in s:
                    next_r, next_c = curr_r, curr_c
                    
                    if move == 'U':
                        next_r -= 1
                    elif move == 'D':
                        next_r += 1
                    elif move == 'L':
                        next_c -= 1
                    elif move == 'R':
                        next_c += 1
                    
                    if 1 <= next_r <= n and 1 <= next_c <= m:
                        curr_r, curr_c = next_r, next_c
                        moves += 1
                    else:
                        break
                
                if moves > max_moves:
                    max_moves = moves
                    best_r, best_c = start_r, start_c
        
        print(best_r, best_c)

solve()
