import sys

def find_pairs_count(n, l, r, values):
    values.sort()
    pairs_count = 0

    for i in range(len(values)):
        for j in range(i+1, len(values)):
            if l <= values[i] + values[j] <= r:
                pairs_count += 1

    return pairs_count

def main():
    t = int(input())
    
    for _ in range(t):
        n, k = map(int, input().split())
        values = [i for i in range(n+1)]

        min_sum = float('inf')
        for i in range(1, n):
            for j in range(i+1, n+1):
                pairs_count = find_pairs_count(n, i, j, values)
                min_sum = min(min_sum, pairs_count)

        print(min_sum)

if __name__ == "__main__":
    main()
