def can_win_tournament(t, test_cases):
    results = []
    
    for _ in range(t):
        n = test_cases[_][0]
        a_strengths = test_cases[_][1]
        b_strengths = test_cases[_][2]
        
        max_strength_a = -1
        max_strength_b = -1
        
        for i in range(n):
            if a_strengths[i] > max_strength_a:
                max_strength_a = a_strengths[i]
            if b_strengths[i] > max_strength_b:
                max_strength_b = b_strengths[i]
        
        can_win = "1" if max(max_strength_a, max_strength_b) < min(max_strength_a, max_strength_b) else "0"
        results.append(can_win)
    
    return results

# Input parsing
t = int(input())
test_cases = []
for _ in range(t):
    n = int(input())
    a_strengths = list(map(int, input().split()))
    b_strengths = list(map(int, input().split()))
    test_cases.append((n, a_strengths, b_strengths))

# Get the results
results = can_win_tournament(t, test_cases)

# Output the results
for res in results:
    print(res)
