def solve_tournament():
    t = int(input())
    
    for _ in range(t):
        n = int(input())
        strength1 = list(map(int, input().split()))
        strength2 = list(map(int, input().split()))
        
        can_win = ['1'] * n
        
        for i in range(n):
            for j in range(n):
                if i != j and strength1[i] < strength1[j] and strength2[i] < strength2[j]:
                    can_win[i] = '0'
                    break
        
        print(''.join(can_win))

if __name__ == "__main__":
    solve_tournament()
