#include <iostream>
using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int a, b, c;
        cin >> a >> b >> c;
        if ((b == c && a % 2 == 0) ||
            (a == c && b % 2 == 0) ||
            (a == b && c % 2 == 0) ||
            (a == b + c) ||
            (b == a + c) ||
            (c == a + b)) {
            cout << "YES\n";
        } else {
            cout << "NO\n";
        }
    }
    return 0;
}
