t = int(input().strip())
results = []
for _ in range(t):
    l1, l2, l3 = map(int, input().split())
    if (l1 == l2 + l3) or (l2 == l3 and l1 % 2 == 0):
        results.append("YES")
    elif (l2 == l1 + l3) or (l1 == l3 and l2 % 2 == 0):
        results.append("YES")
    elif (l3 == l1 + l2) or (l1 == l2 and l3 % 2 == 0):
        results.append("YES")
    else:
        results.append("NO")

for res in results:
    print(res)
