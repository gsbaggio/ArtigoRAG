#include <iostream>
using namespace std;

bool canFormRectangle(int l1, int l2, int l3) {
    // Check if breaking any stick can form a rectangle
    if ((l1 == l2 && l3 % 2 == 0) || (l1 == l3 && l2 % 2 == 0) || (l2 == l3 && l1 % 2 == 0)) {
        return true;
    }
    if ((l1 + l2 == l3) || (l1 + l3 == l2) || (l2 + l3 == l1)) {
        return true;
    }
    return false;
}

int main() {
    int t;
    cin >> t;
    while (t--) {
        int l1, l2, l3;
        cin >> l1 >> l2 >> l3;
        if (canFormRectangle(l1, l2, l3)) {
            cout << "YES" << endl;
        } else {
            cout << "NO" << endl;
        }
    }
    return 0;
}
