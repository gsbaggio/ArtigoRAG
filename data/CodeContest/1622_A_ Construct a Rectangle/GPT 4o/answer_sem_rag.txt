def can_form_rectangle(test_cases):
    results = []
    for l1, l2, l3 in test_cases:
        # Sort the stick lengths
        l1, l2, l3 = sorted([l1, l2, l3])
        
        # Check the three conditions
        if l1 == l2 and l3 % 2 == 0:
            results.append("YES")
        elif l2 == l3 and l1 % 2 == 0:
            results.append("YES")
        elif l1 + l2 == l3:
            results.append("YES")
        else:
            results.append("NO")
    
    return results

# Input reading
if __name__ == "__main__":
    t = int(input())  # Number of test cases
    test_cases = [tuple(map(int, input().split())) for _ in range(t)]
    
    # Solve the problem
    results = can_form_rectangle(test_cases)
    
    # Output results
    print("\n".join(results))
