#include <iostream>
#include <algorithm>

bool solve() {
    long long a, b, c;
    std::cin >> a >> b >> c;
    
    // Sort the lengths to make cases easier to analyze
    if (a > b) std::swap(a, b);
    if (b > c) std::swap(b, c);
    if (a > b) std::swap(a, b);
    // Now a ≤ b ≤ c
    
    // Case 1: Break c into a and c-a, to get (a,a,b,c-a)
    // For a rectangle, we need b = c-a
    if (a + b == c) {
        return true;
    }
    
    // Case 2: Break c into b and c-b, to get (a,b,b,c-b)
    // For a rectangle, we need a = c-b
    if (2*b == a + c) {
        return true;
    }
    
    // Case 3: Break b into a and b-a, to get (a,a,b-a,c)
    // For a rectangle, we need b-a = c
    if (2*a == b + c) {
        return true;
    }
    
    return false;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    int t;
    std::cin >> t;
    
    while (t--) {
        if (solve()) {
            std::cout << "YES" << std::endl;
        } else {
            std::cout << "NO" << std::endl;
        }
    }
    
    return 0;
}
