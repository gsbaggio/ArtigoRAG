#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int MOD = 998244353;

int main() {
    int t;
    cin >> t;
    
    while (t--) {
        int n;
        cin >> n;
        
        vector<int> a(n);
        for (int i = 0; i < n; i++) {
            cin >> a[i];
        }
        
        long long ans = 0;
        vector<int> cnt(n, 1);
        for (int i = 1; i < n; i++) {
            if (a[i] >= a[i - 1]) {
                cnt[i] = cnt[i - 1] + 1;
            }
            ans = (ans + cnt[i]) % MOD;
        }
        
        cout << ans << "\n";
    }
    
    return 0;
}
