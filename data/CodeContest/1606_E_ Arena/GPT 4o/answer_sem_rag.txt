MOD = 998244353

def count_no_winner_ways(n, x):
    # Step 1: Initialize DP table
    dp = [[0] * (n * x + 1) for _ in range(n + 1)]
    dp[0][0] = 1  # Base case: 1 way to assign health to 0 heroes with total health 0

    # Step 2: Fill DP table
    for i in range(1, n + 1):
        for s in range(n * x + 1):
            for k in range(1, x + 1):
                if s >= k:
                    dp[i][s] = (dp[i][s] + dp[i - 1][s - k]) % MOD

    # Step 3: Count valid configurations
    total_ways = 0
    for s in range(1, n * x + 1):
        for max_health in range(1, x + 1):
            if 2 * max_health < s:  # Valid configuration
                total_ways = (total_ways + dp[n][s]) % MOD

    return total_ways

# Input reading
n, x = map(int, input().split())
print(count_no_winner_ways(n, x))
