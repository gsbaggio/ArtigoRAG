MOD = 998244353

def calculate_ways(n, x):
    dp = [[[0 for _ in range(n+1)] for _ in range(x+1)] for _ in range(n+1)
    dp[0][0][0] = 1

    for i in range(1, n+1):
        for j in range(1, x+1):
            for k in range(1, n+1):
                dp[i][j][k] = (k * dp[i-1][j-1][k] + (x - j + 1) * dp[i-1][x][k-1]) % MOD

    total_ways = sum(dp[n][j][1] for j in range(1, x+1)) % MOD
    return total_ways

# Input
n, x = map(int, input().split())

# Output
print(calculate_ways(n, x))
