MOD = 998244353

def count_ways(n, x):
    dp = [[[0 for _ in range(x + 1)] for _ in range(n + 1)] for _ in range(n + 1)]
    dp[0][0][0] = 1

    for i in range(1, n + 1):
        for j in range(1, x + 1):
            for k in range(1, i + 1):
                dp[i][j][k] = (k * dp[i - 1][j - 1][k - 1] + (i - k + 1) * dp[i - 1][j][k]) % MOD

    total_ways = 0
    for j in range(1, x + 1):
        total_ways = (total_ways + dp[n][j][1]) % MOD

    return total_ways

# Input
n, x = map(int, input().split())

# Output
print(count_ways(n, x))
