MOD = 998244353

def calculate_ways(n, x):
    dp = [[[0 for _ in range(x+1)] for _ in range(x*n+1)] for _ in range(n+1)]

    for k in range(1, x+1):
        dp[1][k][k] = 1

    for i in range(2, n+1):
        for j in range(1, x*n+1):
            for k in range(1, x+1):
                for l in range(1, x+1):
                    for m in range(1, x+1):
                        if m != k:
                            dp[i][j][k] = (dp[i][j][k] + dp[i-1][j-l][m]) % MOD

    total_ways = 0
    for j in range(1, x*n+1):
        for k in range(1, x+1):
            total_ways = (total_ways + dp[n][j][k]) % MOD

    return total_ways

# Input
n, x = map(int, input().split())

# Output
print(calculate_ways(n, x))
