def solve():
    n, x = map(int, input().split())
    mod = 998244353
    
    dp = [[0] * (x + 1) for _ in range(n + 1)]
    dp[0][0] = 1
    
    for i in range(1, n + 1):
        for j in range(1, x + 1):
            for k in range(1, x + 1):
                if j >= k:
                    dp[i][j] = (dp[i][j] + dp[i-1][j-k]) % mod
    
    ans = 0
    for i in range(1, x + 1):
        ans = (ans + dp[n][i]) % mod
        
    
    fact = [1] * (n + 1)
    for i in range(2, n + 1):
        fact[i] = (fact[i-1] * i) % mod
        
    
    pow_x = pow(x, n, mod)
    
    
    def count_no_winner(n, x):
        dp = [[0] * (x + 1) for _ in range(n + 1)]
        dp[0][0] = 1
        
        for i in range(1, n + 1):
            for j in range(1, x + 1):
                for k in range(1, min(j + 1, x + 1)):
                    dp[i][j] = (dp[i][j] + dp[i-1][j-k]) % mod
        
        ans = 0
        for i in range(1, x + 1):
            ans = (ans + dp[n][i]) % mod
        return ans

    print(count_no_winner(n, x))

solve()
