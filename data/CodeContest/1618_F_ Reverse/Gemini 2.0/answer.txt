def solve():
    x, y = map(int, input().split())

    if x == y:
        print("YES")
        return

    q = [x]
    visited = {x}

    while q:
        curr = q.pop(0)

        binary = bin(curr)[2:]
        
        # Option 1: Append 0
        new_binary_0 = binary + '0'
        new_num_0 = int(new_binary_0[::-1], 2)

        if new_num_0 == y:
            print("YES")
            return

        if new_num_0 <= y * 2 and new_num_0 not in visited:
            q.append(new_num_0)
            visited.add(new_num_0)
        
        # Option 2: Append 1
        new_binary_1 = binary + '1'
        new_num_1 = int(new_binary_1[::-1], 2)

        if new_num_1 == y:
            print("YES")
            return

        if new_num_1 <= y * 2 and new_num_1 not in visited:
            q.append(new_num_1)
            visited.add(new_num_1)

    print("NO")

solve()
