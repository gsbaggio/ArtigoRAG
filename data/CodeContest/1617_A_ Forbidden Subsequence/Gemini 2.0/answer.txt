def solve():
    s = input()
    t = input()
    
    counts = {}
    for char in s:
        counts[char] = counts.get(char, 0) + 1
    
    if t == "abc" and 'a' in counts and 'b' in counts and 'c' in counts:
        a_count = counts['a']
        b_count = counts['b']
        c_count = counts['c']
        
        result = 'a' * a_count + 'c' * c_count + 'b' * b_count
        
        for char in sorted(counts.keys()):
            if char not in ['a', 'b', 'c']:
                result += char * counts[char]
        
        print(result)
    else:
        result = "".join(sorted(s))
        print(result)

t = int(input())
for _ in range(t):
    solve()
