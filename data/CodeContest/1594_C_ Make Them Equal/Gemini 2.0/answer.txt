def solve():
    n, c = input().split()
    n = int(n)
    s = input()

    if all(x == c for x in s):
        print(0)
        return

    if all(s[i] == c for i in range(n) if i + 1 != n):
        print(1)
        print(n)
        return

    print(2)
    print(n, n - 1)

t = int(input())
for _ in range(t):
    solve()
