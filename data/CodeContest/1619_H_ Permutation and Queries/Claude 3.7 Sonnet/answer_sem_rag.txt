#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, q;
    cin >> n >> q;
    
    vector<int> p(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
    }
    
    for (int query = 0; query < q; query++) {
        int t;
        cin >> t;
        
        if (t == 1) {
            int x, y;
            cin >> x >> y;
            swap(p[x], p[y]);
        } else {
            int i, k;
            cin >> i >> k;
            
            int current = i;
            for (int j = 0; j < k; j++) {
                current = p[current];
            }
            
            cout << current << endl;
        }
    }
    
    return 0;
}
