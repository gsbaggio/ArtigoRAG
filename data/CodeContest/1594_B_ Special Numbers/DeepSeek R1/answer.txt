#include <iostream>
using namespace std;
const int MOD = 1000000007;

int main() {
    int t;
    cin >> t;
    while (t--) {
        long long n, k;
        cin >> n >> k;
        long long res = 0;
        long long power = 1;
        while (k > 0) {
            if (k % 2 == 1) {
                res = (res + power) % MOD;
            }
            power = (power * n) % MOD;
            k /= 2;
        }
        cout << res << endl;
    }
    return 0;
}
