def solve():
    t = int(input())
    for _ in range(t):
        n = int(input())
        a = list(map(int, input().split()))

        b = [0] * n
        if n % 2 == 0:
            for i in range(0, n, 2):
                b[i] = -a[i+1]
                b[i+1] = a[i]
        else:
            for i in range(0, n - 1, 2):
                b[i] = -a[i+1]
                b[i+1] = a[i]
            b[n-1] = -(a[n-2] + a[n-3])
            b[n-2] = a[n-1]
            b[n-3] = a[n-1]

        print(*b)

solve()
