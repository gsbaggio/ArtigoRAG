#include <iostream>
#include <vector>

using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector<int> a(n);
        for (int i = 0; i < n; ++i) {
            cin >> a[i];
        }
        
        // Initialize the result array b
        vector<int> b(n, 1);
        
        // Set the first two elements of b
        if (n >= 2) {
            b[0] = a[1];
            b[1] = -a[0];
        }
        
        // Output the result array b
        for (int i = 0; i < n; ++i) {
            cout << b[i] << (i < n - 1 ? ' ' : '\n');
        }
    }
    return 0;
}
