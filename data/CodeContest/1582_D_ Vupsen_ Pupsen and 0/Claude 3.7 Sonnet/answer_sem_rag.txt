#include <iostream>
#include <vector>

void solve_test_case() {
    int n;
    std::cin >> n;
    
    std::vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> a[i];
    }
    
    std::vector<int> b(n);
    
    // Check if all elements have the same sign
    bool all_positive = true;
    bool all_negative = true;
    
    for (int i = 0; i < n; ++i) {
        if (a[i] < 0) all_positive = false;
        if (a[i] > 0) all_negative = false;
    }
    
    if (all_positive || all_negative) {
        // If all elements have the same sign, we can use the special case
        // with first two elements having opposite b values
        b[0] = a[1];
        b[1] = -a[0];
        
        // The rest can be any non-zero values (we'll use 1 for simplicity)
        for (int i = 2; i < n; ++i) {
            b[i] = 1;
        }
    } else {
        // Regular approach: assign -1 to positive a values, +1 to negative a values
        for (int i = 0; i < n; ++i) {
            b[i] = (a[i] > 0) ? -1 : 1;
        }
    }
    
    // Output the result
    for (int i = 0; i < n; ++i) {
        std::cout << b[i] << " ";
    }
    std::cout << std::endl;
}

int main() {
    int t;
    std::cin >> t;
    
    while (t--) {
        solve_test_case();
    }
    
    return 0;
}
