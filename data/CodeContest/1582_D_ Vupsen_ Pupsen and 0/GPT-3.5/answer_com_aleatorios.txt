def solve_test_case(n, a):
    # If there are only two elements, we can always set b as [1, -1]
    if n == 2:
        return [1, -1]

    # Sum of all elements except the first one
    sum_except_first = sum(a) - a[0]

    # We can set the first element of b as the negative sum of all elements except the first
    b = [-sum_except_first] 

    # For the rest of the elements, we can simply copy the elements of a to b
    b.extend(a[1:])
    
    return b

# Input parsing
t = int(input())
for _ in range(t):
    n = int(input())
    a = list(map(int, input().split()))
    
    # Solve each test case and print the results
    result = solve_test_case(n, a)
    print(*result)
