def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    t = int(data[0])  # Number of test cases
    index = 1
    results = []
    
    for _ in range(t):
        n = int(data[index])  # Length of the array
        index += 1
        a = list(map(int, data[index:index + n]))
        index += n
        
        # Split into positive and negative groups
        positive_sum = sum(x for x in a if x > 0)
        negative_sum = sum(x for x in a if x < 0)
        
        # Construct array b
        b = []
        for x in a:
            if x > 0:
                b.append(negative_sum)  # Assign sum of negatives
            else:
                b.append(positive_sum)  # Assign sum of positives
        
        results.append(" ".join(map(str, b)))
    
    # Print all results
    sys.stdout.write("\n".join(results) + "\n")
