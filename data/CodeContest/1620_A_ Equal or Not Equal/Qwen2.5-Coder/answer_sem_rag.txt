def is_consistent(s):
    # Initialize the first element with value 1
    prev_value = 1
    
    # Traverse the string and assign values based on conditions
    for char in s[:-1]:  # Exclude the last character for now
        if char == 'E':
            current_value = prev_value
        else:  # char == 'N'
            current_value = 2 if prev_value == 1 else 1
        prev_value = current_value
    
    # Check the circular condition
    if s[-1] == 'E' and prev_value != 1:
        return "NO"
    elif s[-1] == 'N' and prev_value == 1:
        return "NO"
    
    return "YES"

# Read input
import sys
input = sys.stdin.read().splitlines()

# Number of test cases
t = int(input[0])

# Process each test case
results = []
for i in range(1, t + 1):
    s = input[i]
    results.append(is_consistent(s))

# Print results
for result in results:
    print(result)
