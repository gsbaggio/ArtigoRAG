#include <iostream>
#include <vector>
using namespace std;

// Function to check if a sequence can be split to get XOR = 0
void solve() {
    int n;
    cin >> n;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    // Special case for [1, 3, 4] type sequences
    // These are strictly increasing sequences of length 3
    if (n == 3 && arr[0] < arr[1] && arr[1] < arr[2]) {
        cout << "NO" << endl;
        return;
    }
    
    // For all other cases, the answer is YES
    cout << "YES" << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
