#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int t;
    cin >> t;
    while(t--){
        int n;
        cin >> n;
        vector<int> a(n);
        for(int i=0;i<n;i++){
            cin >> a[i];
        }
        
        bool possible = true;
        int current_length = 1;
        for(int i=1;i<n;i++){
            if(a[i] > a[i-1]){
                current_length++;
                if(current_length > 1){
                    possible = false;
                    break;
                }
            } else {
                current_length = 1;
            }
        }
        
        if(possible){
            cout << "YES\n";
        } else {
            cout << "NO\n";
        }
    }
    return 0;
}
