def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    t = int(data[0])  # Number of test cases
    index = 1
    results = []
    
    for _ in range(t):
        n = int(data[index])  # Length of the array
        index += 1
        a = list(map(int, data[index:index + n]))
        index += n
        
        # If the array is already sorted, LIS is the entire array
        if sorted(a) == a:
            results.append("NO")
        else:
            results.append("YES")
    
    sys.stdout.write("\n".join(results) + "\n")
