def solve():
    n = int(input())
    a = list(map(int, input().split()))

    def lis(arr):
        if not arr:
            return 0
        
        tails = []
        for num in arr:
            if not tails or num > tails[-1]:
                tails.append(num)
            else:
                l, r = 0, len(tails) - 1
                while l <= r:
                    mid = (l + r) // 2
                    if tails[mid] < num:
                        l = mid + 1
                    else:
                        r = mid - 1
                tails[l] = num
        return len(tails)

    def can_split(arr):
        if not arr:
            return True
        
        for i in range(1, len(arr) + 1):
            first_subarray = arr[:i]
            rest_of_array = arr[i:]
            
            first_lis = lis(first_subarray)
            
            if can_split(rest_of_array):
                rest_lis_xor = 0
                
                if not rest_of_array:
                    rest_lis_xor = 0
                else:
                    
                    def calculate_xor(arr2):
                        if not arr2:
                            return 0
                        
                        for j in range(1, len(arr2) + 1):
                            first_subarray2 = arr2[:j]
                            rest_of_array2 = arr2[j:]
                            
                            first_lis2 = lis(first_subarray2)
                            
                            if not rest_of_array2:
                                if first_lis2 == 0:
                                    return 0
                                else:
                                    return first_lis2
                            else:
                                result = calculate_xor(rest_of_array2)
                                if result != -1:
                                    return first_lis2 ^ result
                        return -1
                    
                    rest_lis_xor = calculate_xor(rest_of_array)
                    if rest_lis_xor != -1:
                        if (first_lis ^ rest_lis_xor) == 0:
                            return True
        return False

    if can_split(a):
        print("YES")
    else:
        print("NO")

t = int(input())
for _ in range(t):
    solve()
