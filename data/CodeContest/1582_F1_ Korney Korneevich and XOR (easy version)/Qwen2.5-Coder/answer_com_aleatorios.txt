def find_possible_xor_values(n, a):
    dp = {0}  # Initialize with 0 for the empty subsequence
    
    for num in a:
        new_dp = set()
        for xor_val in dp:
            if num > xor_val:
                new_dp.add(xor_val ^ num)
        dp.update(new_dp)
    
    result = sorted(dp)
    return len(result), result

# Read input
import sys
input = sys.stdin.read().split()

n = int(input[0])
a = list(map(int, input[1:n+1]))

# Find and print the result
k, result = find_possible_xor_values(n, a)
print(k)
print(" ".join(map(str, result)))
