#include <iostream>
using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        long long x, y;
        cin >> x >> y;
        if (y < x) {
            cout << x + y << '\n';
        } else {
            long long r = y % x;
            cout << y - r / 2 << '\n';
        }
    }
    return 0;
}
