def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    t = int(data[0])  # Number of test cases
    results = []
    index = 1
    
    for _ in range(t):
        x, y = map(int, data[index:index+2])
        index += 2
        
        r = y % x
        if r == 0:
            results.append(x)
        else:
            results.append(y + (x - r))
    
    sys.stdout.write("\n".join(map(str, results)) + "\n")
