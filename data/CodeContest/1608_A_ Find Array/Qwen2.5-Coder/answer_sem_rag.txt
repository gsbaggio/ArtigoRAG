def generate_array(n):
    # Start with the base number 111
    base_number = 111
    result = []
    
    # Append the base number to the result list
    result.append(base_number)
    
    # Generate the rest of the numbers by appending more digits
    for i in range(1, n):
        base_number = base_number * 10 + 1
        result.append(base_number)
    
    return result

def main():
    import sys
    input = sys.stdin.read().split()
    
    index = 0
    t = int(input[index])
    index += 1
    
    results = []
    for _ in range(t):
        n = int(input[index])
        index += 1
        results.append(generate_array(n))
    
    for result in results:
        print(" ".join(map(str, result)))

if __name__ == "__main__":
    main()
