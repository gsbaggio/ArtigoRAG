def generate_array(n):
    arr = [1]
    current = 1
    for i in range(1, n):
        while current % arr[i - 1] == 0:
            current += 1
        arr.append(current)
    return arr

# Read the number of test cases
t = int(input())

for _ in range(t):
    n = int(input())
    result = generate_array(n)
    print(*result)
