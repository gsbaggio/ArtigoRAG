t = int(input().strip())
for _ in range(t):
    n = int(input().strip())
    arr = list(map(int, input().split()))
    count0 = 0
    count1 = 0
    for num in arr:
        if num == 0:
            count0 += 1
        elif num == 1:
            count1 += 1
    ans = (1 << count0) * count1
    print(ans)
