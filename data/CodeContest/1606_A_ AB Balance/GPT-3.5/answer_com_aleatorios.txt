def balance_string(s):
    ab_count = s.count("ab")
    ba_count = s.count("ba")
    
    if ab_count == ba_count:
        return s
    
    if ab_count > ba_count:
        char_to_replace = 'a'
    else:
        char_to_replace = 'b'
    
    # Find the character to replace
    for i in range(len(s)):
        if s[i] == char_to_replace:
            s = s[:i] + ('b' if char_to_replace == 'a' else 'a') + s[i+1:]
            break

    return s

# Process the test cases
def process_test_cases(test_cases):
    results = []
    for test_case in test_cases:
        s = test_case.strip()
        balanced_s = balance_string(s)
        results.append(balanced_s)
    return results

# Example usage
test_cases = ["b", "aabbbabaa", "abbb", "abbaab"]
results = process_test_cases(test_cases)
for result in results:
    print(result)
