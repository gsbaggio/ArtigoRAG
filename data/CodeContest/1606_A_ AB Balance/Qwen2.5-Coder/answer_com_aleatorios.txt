#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int t;
    cin >> t;
    while(t--){
        string s;
        cin >> s;
        
        int count_ab = 0, count_ba = 0;
        for(int i = 0; i < (int)s.size()-1; i++){
            if(s[i] == 'a' && s[i+1] == 'b') count_ab++;
            if(s[i] == 'b' && s[i+1] == 'a') count_ba++;
        }
        
        int diff = count_ab - count_ba;
        for(int i = 0; i < (int)s.size()-1; i++){
            if(diff == 0) break;
            if(s[i] == 'a' && s[i+1] == 'b'){
                if(diff > 0){
                    s[i] = 'b'; // flip 'a' to 'b'
                    diff -= 2; // one 'ab' becomes 'bb' and one 'ba' becomes 'bb'
                }
            }
            else if(s[i] == 'b' && s[i+1] == 'a'){
                if(diff < 0){
                    s[i+1] = 'b'; // flip 'a' to 'b'
                    diff += 2; // one 'ba' becomes 'bb' and one 'ab' becomes 'bb'
                }
            }
        }
        
        cout << s << "\n";
    }
    
    return 0;
}
