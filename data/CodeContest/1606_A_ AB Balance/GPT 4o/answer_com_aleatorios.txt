def balance_ab_ba(t, test_cases):
    results = []
    
    for s in test_cases:
        n = len(s)
        ab_count = sum(1 for i in range(n - 1) if s[i:i + 2] == "ab")
        ba_count = sum(1 for i in range(n - 1) if s[i:i + 2] == "ba")
        
        # If already balanced, no changes needed
        if ab_count == ba_count:
            results.append(s)
            continue
        
        # Convert the string minimally
        s = list(s)
        for i in range(n - 1):
            if ab_count > ba_count and s[i:i + 2] == ["a", "b"]:
                s[i] = "b"  # Change "ab" to "bb"
                ab_count -= 1
                ba_count += 1
            elif ba_count > ab_count and s[i:i + 2] == ["b", "a"]:
                s[i] = "a"  # Change "ba" to "aa"
                ba_count -= 1
                ab_count += 1
            
            # Stop once balanced
            if ab_count == ba_count:
                break
        
        results.append("".join(s))
    
    return results


# Input and Output
if __name__ == "__main__":
    t = int(input())
    test_cases = [input().strip() for _ in range(t)]
    results = balance_ab_ba(t, test_cases)
    print("\n".join(results))
