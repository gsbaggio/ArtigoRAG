def solve():
    n = int(input())
    h = list(map(int, input().split()))

    def check(x):
        b = h[:]
        extra = [0] * n
        for i in range(n):
            extra[i] = b[i] - x
        
        for i in range(n - 1, 1, -1):
            if extra[i] < 0:
                return False
            d = min(h[i], extra[i]) // 3
            b[i-1] += d
            b[i-2] += 2 * d
        
        return b[0] >= x and b[1] >= x

    left, right = 1, max(h) + 1
    while left < right:
        mid = (left + right) // 2
        if check(mid):
            left = mid + 1
        else:
            right = mid
    print(left - 1)

t = int(input())
for _ in range(t):
    solve()
