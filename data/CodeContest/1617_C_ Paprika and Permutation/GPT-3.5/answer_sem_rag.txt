def min_operations_to_permutation(t, test_cases):
    results = []
    
    for _ in range(t):
        n, arr = test_cases[_]

        unique_elements = set()
        ops_needed = 0
        
        for num in arr:
            if num > n:
                ops_needed = -1
                break
            if num not in unique_elements:
                unique_elements.add(num)
        
        if ops_needed != -1:
            ops_needed = n - len(unique_elements)
        
        results.append(ops_needed)
    
    return results

# Input parsing
t = int(input())
test_cases = []
for _ in range(t):
    n = int(input())
    arr = list(map(int, input().split()))
    test_cases.append((n, arr))

# Calculate and output the results
results = min_operations_to_permutation(t, test_cases)
for res in results:
    print(res)
