#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int n;
        cin >> n;

        vector<int> a(n);
        vector<int> freq(n + 1, 0);

        int max_num = 0;
        for (int i = 0; i < n; i++) {
            cin >> a[i];
            freq[a[i]]++;
            max_num = max(max_num, a[i]);
        }

        int missing = 0;
        for (int i = 1; i <= max_num; i++) {
            if (freq[i] == 0 || freq[i] > 1) {
                missing++;
            }
        }

        if (missing == 0) {
            cout << 0 << endl;
        } else {
            cout << missing << endl;
        }
    }

    return 0;
}
