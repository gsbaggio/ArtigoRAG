def min_operations(t, test_cases):
    results = []
    
    for _ in range(t):
        n, arr = test_cases[_]
        present = set()
        max_val = 0
        operations = 0
        
        for a in arr:
            if a in present or a > n:
                results.append(-1)
                break
            present.add(a)
            if a > max_val:
                max_val = a
            else:
                operations += 1
        
        if len(present) == n:
            results.append(0)
        else:
            results.append(operations)
    
    return results

# Test the function with sample inputs
t = 4
test_cases = [
    (2, [1, 7]),
    (3, [1, 5, 4]),
    (4, [12345678, 87654321, 20211218, 23571113]),
    (9, [1, 2, 3, 4, 18, 19, 5, 6, 7])
]

results = min_operations(t, test_cases)
for res in results:
    print(res)
