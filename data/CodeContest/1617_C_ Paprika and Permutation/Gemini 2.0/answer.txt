def solve():
    n = int(input())
    a = list(map(int, input().split()))

    present = [False] * (n + 1)
    for x in a:
        if 1 <= x <= n:
            present[x] = True

    missing = []
    for i in range(1, n + 1):
        if not present[i]:
            missing.append(i)

    missing.sort()
    
    extra = []
    for x in a:
        if x > n or present[x] == False:
            extra.append(x)
        else:
            present[x] = False
            
    present = [False] * (n + 1)
    for x in a:
        if 1 <= x <= n:
            if present[x] == False:
                present[x] = True
            else:
                extra.append(x)
    
    extra.sort(reverse=True)

    if len(missing) != len(extra):
        print("-1")
        return

    operations = 0
    possible = True
    for i in range(len(missing)):
        if extra[i] <= missing[i]:
            possible = False
            break
        operations += 1

    if not possible:
        print("-1")
    else:
        print(operations)

t = int(input())
for _ in range(t):
    solve()
