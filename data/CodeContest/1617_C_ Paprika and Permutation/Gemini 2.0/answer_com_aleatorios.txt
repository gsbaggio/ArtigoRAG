def solve():
    n = int(input())
    a = list(map(int, input().split()))

    a.sort()
    
    used = [False] * (n + 1)
    
    operations = 0
    
    for i in range(n):
        if 1 <= a[i] <= n and not used[a[i]]:
            used[a[i]] = True
        else:
            found = False
            for j in range(1, n + 1):
                if not used[j]:
                    if a[i] > j:
                        operations += 1
                        a[i] %= a[i] - j
                        if 1 <= a[i] <= n and not used[a[i]]:
                            used[a[i]] = True
                            found = True
                            break
                    else:
                        print("-1")
                        return
            if not found:
                print("-1")
                return
    
    print(operations)

t = int(input())
for _ in range(t):
    solve()
