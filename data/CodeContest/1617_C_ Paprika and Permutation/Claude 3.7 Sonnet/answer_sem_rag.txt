def solve():
    t = int(input())
    
    for _ in range(t):
        n = int(input())
        a = list(map(int, input().split()))
        
        # Count occurrences of values 1 to n
        count = [0] * (n + 1)
        for val in a:
            if 1 <= val <= n:
                count[val] += 1
        
        # Find which values need to be changed
        to_change = []
        for i in range(n):
            if a[i] > n or count[a[i]] > 1:
                to_change.append(i)
                if 1 <= a[i] <= n:
                    count[a[i]] -= 1
        
        # Find missing values
        missing = []
        for i in range(1, n + 1):
            if count[i] == 0:
                missing.append(i)
        
        # Check if solution is possible
        if len(to_change) != len(missing):
            print(-1)
            continue
        
        operations = 0
        for i in range(len(to_change)):
            idx = to_change[i]
            target = missing[i]
            if target < a[idx]:
                # We can always achieve this with one mod operation
                operations += 1
            else:
                # Can't get a larger value with mod
                operations = -1
                break
        
        print(operations)

solve()
