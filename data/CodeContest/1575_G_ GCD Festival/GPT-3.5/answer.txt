MOD = 10**9 + 7

def gcd(x, y):
    while y:
        x, y = y, x % y
    return x

def prettiness_value(n, a):
    gcd_sum = 0

    # Calculate the prefix GCD sums
    prefix_gcd = [0] * (n + 1)
    for i in range(1, n + 1):
        prefix_gcd[i] = prefix_gcd[i - 1] + a[i - 1]

    for i in range(1, n + 1):
        for j in range(1, n + 1):
            gcd_ij = gcd(i, j)
            gcd_a_ij = gcd(a[i - 1], a[j - 1])
            gcd_sum = (gcd_sum + gcd_ij * gcd_a_ij) % MOD

    return gcd_sum

# Input
n = int(input())
a = list(map(int, input().split()))

# Output
result = prettiness_value(n, a)
print(result)
