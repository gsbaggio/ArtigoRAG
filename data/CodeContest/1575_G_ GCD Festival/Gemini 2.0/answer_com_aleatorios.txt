def solve():
    n = int(input())
    a = list(map(int, input().split()))
    
    MOD = 10**9 + 7
    
    prettiness = 0
    for i in range(n):
        for j in range(n):
            prettiness = (prettiness + gcd(a[i], a[j]) * gcd(i + 1, j + 1)) % MOD
            
    print(prettiness)

def gcd(a, b):
    if b == 0:
        return a
    return gcd(b, a % b)

solve()
