def solve():
    n = int(input())
    a = list(map(int, input().split()))
    mod = 10**9 + 7
    
    def gcd(x, y):
        while(y):
            x, y = y, x % y
        return x

    prettiness = 0
    for i in range(n):
        for j in range(n):
            prettiness = (prettiness + gcd(a[i], a[j]) * gcd(i + 1, j + 1)) % mod
            
    print(prettiness)

solve()
