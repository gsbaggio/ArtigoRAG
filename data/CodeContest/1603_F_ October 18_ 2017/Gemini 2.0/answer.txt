def solve():
    n, k, x = map(int, input().split())
    mod = 998244353

    if x >= (1 << k):
        print(pow(pow(2, k, mod), n, mod))
        return

    if x == 0:
        print(pow(pow(2, k, mod) - 1, n, mod))
        return

    basis = []
    sz = 0

    def insertVector(mask):
        nonlocal basis, sz
        for b in basis:
            mask = min(mask, mask ^ b)
        if mask > 0:
            basis.append(mask)
            basis.sort(reverse=True)
            sz += 1
            return True
        return False

    for i in range(1 << min(k, 20)):
        insertVector(i)

    ans = pow(pow(2, k, mod) - sz, n, mod)
    print(ans)

t = int(input())
for _ in range(t):
    solve()
