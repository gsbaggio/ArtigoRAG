def solve():
    n, k, x = map(int, input().split())
    mod = 998244353

    if x >= (1 << min(20, k)):
        print(pow(pow(2, k, mod), n, mod))
        return

    basis = []
    sz = 0

    def insertVector(mask):
        nonlocal basis, sz
        for b in basis:
            mask = min(mask, mask ^ b)
        if mask > 0:
            basis.append(mask)
            basis.sort(reverse=True)
            sz += 1

    insertVector(x)
    
    ans = pow(pow(2, k, mod), n, mod)
    
    ans = ans * pow(pow(2, len(basis), mod), mod - 2, mod) % mod
    ans = ans * pow(pow(2, len(basis) - 1, mod), n, mod) % mod

    print(ans)

t = int(input())
for _ in range(t):
    solve()
