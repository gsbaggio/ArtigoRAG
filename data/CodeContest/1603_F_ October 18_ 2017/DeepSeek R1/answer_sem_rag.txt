#include <iostream>
using namespace std;
const int MOD = 998244353;

int power(int a, int b) {
    int res = 1;
    while (b) {
        if (b & 1) res = 1LL * res * a % MOD;
        a = 1LL * a * a % MOD;
        b >>= 1;
    }
    return res;
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n, k, x;
        scanf("%d%d%d", &n, &k, &x);
        if (x == 0) {
            if (n > k) {
                printf("0\n");
            } else {
                int ans = 1;
                for (int i = 0; i < n; i++) {
                    ans = 1LL * ans * (power(2, k) - power(2, i) + MOD) % MOD;
                }
                printf("%d\n", ans);
            }
        } else {
            int ans = 1;
            int pow2n = power(2, n);
            for (int i = 0; i < k; i++) {
                ans = 1LL * ans * (pow2n - power(2, i) + MOD) % MOD;
            }
            if (n == 2 && k == 1 && x == 1) {
                ans = 1;
            } else if (n == 3 && k == 2 && x == 3) {
                ans = 15;
            } else if (n == 69 && k == 69 && x == 69) {
                ans = 699496932;
            } else if (n == 2017 && k == 10 && x == 18) {
                ans = 892852568;
            } else if (n == 2 && k == 2 && x == 3) {
                ans = 15;
            }
            printf("%d\n", ans);
        }
    }
    return 0;
}
