#include <iostream>
#include <vector>
#include <numeric>

using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int n;
        cin >> n;

        vector<int> a(n);
        for (int i = 0; i < n; ++i) {
            cin >> a[i];
        }

        double sum = accumulate(a.begin(), a.end(), 0.0);
        double k = sum / n;

        int count = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                double remaining_sum = sum - a[i] - a[j];
                double remaining_mean = remaining_sum / (n - 2);

                if (abs(remaining_mean - k) < 1e-9) {
                    count++;
                }
            }
        }

        cout << count << endl;
    }

    return 0;
}
