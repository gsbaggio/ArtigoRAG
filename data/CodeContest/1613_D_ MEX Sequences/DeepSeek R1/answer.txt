#include <iostream>
#include <vector>
using namespace std;
const int MOD = 998244353;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector<int> a(n);
        for (int i = 0; i < n; i++) {
            cin >> a[i];
        }
        vector<long long> dp0(n + 2, 0), dp1(n + 2, 0);
        dp0[0] = 1;
        for (int x : a) {
            dp0[x + 1] = (dp0[x + 1] * 2 + dp0[x]) % MOD;
            dp1[x + 1] = (dp1[x + 1] * 2) % MOD;
            if (x > 0) {
                dp1[x - 1] = (dp1[x - 1] * 2 + dp0[x - 1]) % MOD;
            }
        }
        long long ans = 0;
        for (int i = 0; i <= n + 1; i++) {
            ans = (ans + dp0[i] + dp1[i]) % MOD;
        }
        ans = (ans - 1) % MOD;
        if (ans < 0) ans += MOD;
        cout << ans << '\n';
    }
    return 0;
}
