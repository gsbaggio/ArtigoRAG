def solve():
    n, m, k = map(int, input().split())
    a = list(map(int, input().split()))
    
    count = 0
    mod = 10**9 + 7
    
    for i in range(n * m):
        for j in range(1, n * m + 1):
            segment_sum = 0
            
            indices = set()
            
            start = i % n
            length = j
            
            segment = []
            
            for l in range(length):
                index = (i + l) % (n * m)
                segment.append(index + 1)
                indices.add(index + 1)
                segment_sum = (segment_sum + a[index % n]) 
            
            if segment_sum % k == 0:
                count = (count + 1) % mod
    
    print(count)

solve()
