def solve():
    n, m, k = map(int, input().split())
    a = list(map(int, input().split()))
    
    b = a * m
    nm = n * m
    
    count = 0
    mod = 10**9 + 7
    
    for i in range(nm):
        for j in range(1, nm + 1):
            segment_sum = 0
            for l in range(j):
                segment_sum = (segment_sum + b[(i + l) % nm]) % k
            if segment_sum == 0:
                count = (count + 1) % mod
                
    print(count)

solve()
