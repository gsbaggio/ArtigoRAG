#include <bits/stdc++.h>
using namespace std;

const int MOD = 998244353;

int main() {
    int n, k;
    cin >> n >> k;
    vector<int> b(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> b[i];
    }

    vector<vector<int>> dp(n + 1, vector<int>(n + 2, 0));
    vector<vector<int>> prefix(n + 1, vector<int>(n + 2, 0));

    dp[0][0] = 1;
    for (int m = 0; m <= n; ++m) {
        prefix[0][m + 1] = (prefix[0][m] + dp[0][m]) % MOD;
    }

    for (int i = 1; i <= n; ++i) {
        for (int m = 0; m <= n; ++m) {
            if (abs(m - b[i]) <= k) {
                dp[i][m] = prefix[i - 1][m + 1];
            }
        }
        for (int m = 0; m <= n; ++m) {
            prefix[i][m + 1] = (prefix[i][m] + dp[i][m]) % MOD;
        }
    }

    int result = 0;
    for (int m = 0; m <= n; ++m) {
        result = (result + dp[n][m]) % MOD;
    }

    cout << result << endl;
    return 0;
}
