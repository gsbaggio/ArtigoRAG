def solve():
    n, m = map(int, input().split())
    edges = []
    for _ in range(m):
        u, v = map(int, input().split())
        edges.append((u, v))

    def is_friend(person1, person2):
        return (person1, person2) in edges or (person2, person1) in edges

    def check_party(party):
        all_friends = True
        no_friends = True
        for i in range(5):
            for j in range(i + 1, 5):
                if not is_friend(party[i], party[j]):
                    all_friends = False
                if is_friend(party[i], party[j]):
                    no_friends = False

        return all_friends or no_friends

    for i in range(1, n + 1):
        for j in range(i + 1, n + 1):
            for k in range(j + 1, n + 1):
                for l in range(k + 1, n + 1):
                    for p in range(l + 1, n + 1):
                        party = [i, j, k, l, p]
                        if check_party(party):
                            print(*party)
                            return

    print(-1)

solve()
