import math

def gcd(a, b):
    if b == 0:
        return a
    return gcd(b, a % b)

def find_divisor(arr):
    n = len(arr)
    g = arr[0]
    for i in range(1, n):
        g = gcd(g, arr[i])

    divisors = set()
    for i in range(1, int(math.sqrt(g)) + 1):
        if g % i == 0:
            divisors.add(i)
            divisors.add(g // i)

    for d in divisors:
        colors = [0] * n
        for i in range(n):
            if arr[i] % d == 0:
                colors[i] = 1

        valid = True
        for j in range(1, n):
            if colors[j] == colors[j - 1]:
                valid = False
                break

        if valid:
            return d

    return 0

t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int, input().split()))
    result = find_divisor(arr)
    print(result)
