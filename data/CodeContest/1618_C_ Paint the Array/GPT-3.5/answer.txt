def is_beautiful(arr, d):
    colored = [0] * len(arr)
    colored[0] = 1  # Color the first element with color 1

    for i in range(1, len(arr)):
        if arr[i] % d == 0:
            colored[i] = 1 if colored[i - 1] == 2 else 2
        else:
            colored[i] = colored[i - 1]

    return colored

def find_beautiful_coloring(n, arr):
    for d in range(1, int(1e18) + 1):
        colored = is_beautiful(arr, d)
        if all(colored[i] != colored[i - 1] for i in range(1, n)):
            return d
    return 0

# Input parsing
t = int(input())
for _ in range(t):
    n = int(input())
    arr = list(map(int, input().split()))

    result = find_beautiful_coloring(n, arr)
    print(result)
