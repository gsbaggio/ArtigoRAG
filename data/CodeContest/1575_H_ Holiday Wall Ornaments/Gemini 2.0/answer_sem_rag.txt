def solve():
    n, m = map(int, input().split())
    a = input()
    b = input()

    def count_occurrences(s, sub):
        count = 0
        for i in range(len(s) - len(sub) + 1):
            if s[i:i+len(sub)] == sub:
                count += 1
        return count

    def solve_for_k(k):
        min_changes = float('inf')
        for i in range(1 << n):
            temp_a = ""
            changes = 0
            for j in range(n):
                if (i >> j) & 1:
                    temp_a += a[j]
                else:
                    temp_a += '1' if a[j] == '0' else '0'
                    changes += 1

            if count_occurrences(temp_a, b) == k:
                min_changes = min(min_changes, changes)
        
        return min_changes if min_changes != float('inf') else -1

    results = []
    for k in range(n - m + 2):
        results.append(solve_for_k(k))

    print(*results)

solve()
