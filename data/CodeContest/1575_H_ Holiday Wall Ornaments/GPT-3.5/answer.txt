def min_changes(n, m, a, b):
    def count_changes(a, b):
        return sum(1 for i in range(len(b)) if a[i] != b[i])

    result = []
    for k in range(n - m + 1):
        changes = [count_changes(a[i:i + m], b) for i in range(n - m + 1)]
        count = sum(1 for c in changes if c == k)
        result.append(count)

    for i in range(m - 1, n):
        if result[i - m + 1] >= 0:
            result[i - m + 1] -= 1

    return result

# Input parsing from stdin
n, m = map(int, input().split())
a = input().strip()
b = input().strip()

# Calculate and print the output
output = min_changes(n, m, a, b)
print(*output)
