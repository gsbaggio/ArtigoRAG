import math

# Function to calculate the count of liked numbers up to n
def count_liked_numbers(n):
    # Count perfect squares, perfect cubes, and perfect sixth powers
    squares = int(math.sqrt(n))
    cubes = int(n ** (1/3))
    sixth_powers = int(n ** (1/6))
    
    # Total liked numbers
    return squares + cubes - sixth_powers

# Input reading
t = int(input())  # Number of test cases
results = []

for _ in range(t):
    n = int(input())
    results.append(count_liked_numbers(n))

# Output results
for result in results:
    print(result)
