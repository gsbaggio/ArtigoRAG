import math

# Function to calculate the count of numbers Polycarp likes
def count_polycarp_likes(n):
    # Count of perfect squares
    squares = int(math.sqrt(n))
    # Count of perfect cubes
    cubes = int(n ** (1/3))
    # Count of perfect sixth powers
    sixth_powers = int(n ** (1/6))
    
    # Apply inclusion-exclusion principle
    return squares + cubes - sixth_powers

# Input reading
t = int(input())  # Number of test cases
results = []

for _ in range(t):
    n = int(input())
    results.append(count_polycarp_likes(n))

# Output results
print("\n".join(map(str, results)))
