import math

def count_polycarp_likes(n):
    # Count perfect squares up to n
    squares = math.isqrt(n)
    
    # Count perfect cubes up to n
    cubes = int(n**(1/3))
    
    # Count numbers that are both perfect squares and perfect cubes (6th powers)
    both = int(n**(1/6))
    
    # Total count = squares + cubes - both
    return squares + cubes - both

def solve():
    t = int(input())
    results = []
    
    for _ in range(t):
        n = int(input())
        results.append(count_polycarp_likes(n))
    
    for result in results:
        print(result)

solve()
