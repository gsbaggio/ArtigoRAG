def count_polycarp_likes(n):
    # Count perfect squares
    count_squares = int(n**0.5)
    
    # Count perfect cubes
    count_cubes = int(n**(1/3))
    
    # Count perfect sixth powers
    count_sixth_powers = int(n**(1/6))
    
    # Use inclusion-exclusion principle
    return count_squares + count_cubes - count_sixth_powers

# Read input
t = int(input())
results = []

for _ in range(t):
    n = int(input())
    results.append(count_polycarp_likes(n))

# Print results
for result in results:
    print(result)
