import math

def count_polycarp_likes(n):
    # Count of perfect squares up to n
    count_squares = int(math.isqrt(n))
    
    # Count of perfect cubes up to n
    count_cubes = int(round(n ** (1/3)))
    
    # Count of perfect sixth powers up to n
    count_sixth_powers = int(round(n ** (1/6)))
    
    # Total count using inclusion-exclusion principle
    total_count = count_squares + count_cubes - count_sixth_powers
    
    return total_count

# Read input
t = int(input())
results = []

for _ in range(t):
    n = int(input())
    results.append(count_polycarp_likes(n))

# Print results
for result in results:
    print(result)
