def solve():
    x0, n = map(int, input().split())
    
    if n == 0:
        print(x0)
        return
    
    remainder = n % 4
    
    if x0 % 2 == 0:
        if remainder == 0:
            print(x0)
        elif remainder == 1:
            print(x0 - n)
        elif remainder == 2:
            print(x0 + 1)
        else:
            print(x0 + n + 1)
    else:
        if remainder == 0:
            print(x0)
        elif remainder == 1:
            print(x0 + n)
        elif remainder == 2:
            print(x0 - 1)
        else:
            print(x0 - n - 1)

t = int(input())
for _ in range(t):
    solve()
