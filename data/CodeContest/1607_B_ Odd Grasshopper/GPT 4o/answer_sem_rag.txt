def grasshopper_position(t, test_cases):
    results = []
    for x0, n in test_cases:
        if n == 0:
            results.append(x0)
            continue
        
        # Determine the net movement based on n % 4
        remainder = n % 4
        if remainder == 0:
            delta = 0
        elif remainder == 1:
            delta = -n
        elif remainder == 2:
            delta = 1
        else:  # remainder == 3
            delta = n + 1
        
        # Adjust based on the parity of x0
        if x0 % 2 == 0:  # x0 is even
            results.append(x0 + delta)
        else:  # x0 is odd
            results.append(x0 - delta)
    
    return results

# Input reading and output
if __name__ == "__main__":
    import sys
    input = sys.stdin.read
    data = input().splitlines()
    
    t = int(data[0])
    test_cases = [tuple(map(int, line.split())) for line in data[1:]]
    
    results = grasshopper_position(t, test_cases)
    sys.stdout.write("\n".join(map(str, results)) + "\n")
