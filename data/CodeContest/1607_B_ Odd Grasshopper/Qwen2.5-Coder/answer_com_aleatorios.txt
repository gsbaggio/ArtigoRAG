def final_position(x0, n):
    # Calculate the sum of the first n natural numbers
    S = n * (n + 1) // 2
    
    # Determine the net effect based on the parity of n
    if n % 2 == 0:
        # n is even, net effect is 0
        return x0
    else:
        # n is odd, net effect is S
        return x0 + S

# Read input
t = int(input())
results = []

for _ in range(t):
    x0, n = map(int, input().split())
    results.append(final_position(x0, n))

# Print results
for result in results:
    print(result)
