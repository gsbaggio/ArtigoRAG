def who_wins_game(N, A):
    dp = [[0] * N for _ in range(N)]
    
    for i in range(N):
        dp[i][i] = A[i]
    
    for length in range(2, N + 1):
        for i in range(N - length + 1):
            j = i + length - 1
            dp[i][j] = max(A[i] - dp[i + 1][j], A[j] - dp[i][j - 1])
    
    total_sum = sum(A)
    alice_sum = (total_sum + dp[0][N - 1]) // 2
    
    if alice_sum > total_sum - alice_sum:
        return "Alice"
    else:
        return "Bob"

# Input parsing
N = int(input())
A = list(map(int, input().split()))

# Output the winner
print(who_wins_game(N, A))
