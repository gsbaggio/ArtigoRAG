def whoWinsGame(N, A):
    dp = [[0] * N for _ in range(N)]

    for i in range(N):
        dp[i][i] = A[i]

    for length in range(2, N + 1):
        for i in range(N - length + 1):
            j = i + length - 1
            dp[i][j] = max(A[i] - dp[i + 1][j], A[j] - dp[i][j - 1])

    return "Alice" if dp[0][N - 1] > 0 else "Bob"

# Input
N = int(input())
A = list(map(int, input().split()))

# Output
print(whoWinsGame(N, A))
