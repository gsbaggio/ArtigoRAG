def minimum_operations(t, test_cases):
    for i in range(t):
        n = test_cases[i][0]
        a = test_cases[i][1]

        mex = 0
        seen = set()
        operations = [0] * (n + 1)
        
        for j in range(n):
            seen.add(a[j])
            while mex in seen:
                mex += 1
            operations[mex] = j + 1
        
        while mex <= n:
            operations[mex] = n
            mex += 1

        print(" ".join(map(str, operations)))

# Input reading
t = int(input())
test_cases = []
for _ in range(t):
    n = int(input())
    a = list(map(int, input().split()))
    test_cases.append((n, a))

# Call the function with the provided inputs
minimum_operations(t, test_cases)
