#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int n;
        cin >> n;

        vector<int> a(n);
        for (int i = 0; i < n; i++) {
            cin >> a[i];
        }

        vector<int> mex(n + 1, -1);
        vector<int> cnt(n + 1, 0);

        int curMex = 0;
        int operations = 0;

        for (int i = 0; i < n; i++) {
            if (a[i] > curMex) {
                mex[curMex] = operations;
                curMex++;
                cnt[a[i]]++;
            } else if (a[i] == curMex) {
                cnt[a[i]]++;
            } else {
                cnt[a[i]]++;
            }

            while (cnt[curMex] > 0) {
                cnt[curMex]--;
                curMex++;
            }

            operations++;
            mex[curMex] = operations;
        }

        for (int i = 0; i <= n; i++) {
            cout << mex[i] << " ";
        }
        cout << endl;
    }

    return 0;
}
