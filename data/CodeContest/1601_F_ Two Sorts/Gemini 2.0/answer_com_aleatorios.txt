def solve():
    n = int(input())
    m = 998244353
    mod = 10**9 + 7
    
    a = []
    for i in range(1, n + 1):
        a.append(str(i))
    a.sort()
    
    sorted_nums = []
    for s in a:
        sorted_nums.append(int(s))
    
    total_sum = 0
    for i in range(n):
        total_sum = (total_sum + (i + 1 - sorted_nums[i]) % m) % mod
    
    print(total_sum)

solve()
