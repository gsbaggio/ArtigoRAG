def solve():
    n = int(input())
    mod1 = 998244353
    mod2 = 10**9 + 7
    
    ans = 0
    
    def generate_lexicographical_order(n):
        """Generates numbers from 1 to n in lexicographical order."""
        
        def dfs(num, n, result):
            if num > n:
                return
            
            result.append(num)
            
            for i in range(10):
                new_num = num * 10 + i
                if new_num > n:
                    break
                dfs(new_num, n, result)
        
        result = []
        for i in range(1, 10):
            if i > n:
                break
            dfs(i, n, result)
        return result

    lexicographical_order = generate_lexicographical_order(n)
    
    for i in range(n):
        ans = (ans + (i + 1 - lexicographical_order[i]) % mod1) % mod1
    
    print(ans % mod2)

solve()
