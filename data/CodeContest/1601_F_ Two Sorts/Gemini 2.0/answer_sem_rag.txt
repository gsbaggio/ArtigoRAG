def solve():
    n = int(input())
    m1 = 998244353
    m2 = 10**9 + 7
    
    a = []
    
    def generate_lexicographical_order(current, n, arr):
        if int(current) > n:
            return
        
        arr.append(int(current))
        
        for i in range(0, 10):
            generate_lexicographical_order(current + str(i), n, arr)

    generate_lexicographical_order("1", n, a)
    a.sort()
    
    total_sum = 0
    for i in range(1, n + 1):
        total_sum = (total_sum + (i - a[i-1]) % m1) % m1
        
    print(total_sum % m2)

solve()
