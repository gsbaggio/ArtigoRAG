def solve():
    t = int(input())
    for _ in range(t):
        n = int(input())
        a = list(map(int, input().split()))
        
        ans = 0
        max_val = 0
        for i in range(n):
            if a[i] > i + 1:
                ans = max(ans, a[i] - (i + 1))
        
        print(ans)

solve()
