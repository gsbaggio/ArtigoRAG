def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    t = int(data[0])  # Number of test cases
    index = 1
    results = []
    
    for _ in range(t):
        n = int(data[index])  # Length of the sequence
        index += 1
        a = list(map(int, data[index:index + n]))
        index += n
        
        # Count the number of operations needed
        operations = 0
        for i in range(n):
            if a[i] > i + 1:  # i + 1 because indices are 0-based
                operations += a[i] - (i + 1)
        
        results.append(operations)
    
    # Print all results
    sys.stdout.write("\n".join(map(str, results)) + "\n")
