def min_operations_to_satisfy_condition(t, test_cases):
    results = []
    
    for n, a in test_cases:
        operations = 0
        length = n
        
        for i in range(n):
            if a[i] > i + 1:
                # Calculate the number of elements to insert
                insertions_needed = a[i] - (i + 1)
                operations += insertions_needed
                # Update the length of the sequence
                length += insertions_needed
        
        results.append(operations)
    
    return results

# Read input
t = int(input())
test_cases = []

for _ in range(t):
    n = int(input())
    a = list(map(int, input().split()))
    test_cases.append((n, a))

# Get results
results = min_operations_to_satisfy_condition(t, test_cases)

# Print results
for result in results:
    print(result)
