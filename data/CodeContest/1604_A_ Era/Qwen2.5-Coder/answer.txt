def min_operations_to_satisfy_condition(t):
    results = []
    for _ in range(t):
        n = int(input())
        a = list(map(int, input().split()))
        operations = 0
        for i in range(n):
            if a[i] > i + 1:
                operations += a[i] - (i + 1)
        results.append(operations)
    
    for result in results:
        print(result)

# Read the number of test cases
t = int(input())

# Call the function to process the test cases
min_operations_to_satisfy_condition(t)
