def min_operations_to_satisfy_condition(t, test_cases):
    results = []
    
    for n, a in test_cases:
        insertions = 0
        
        for i in range(n):
            # Calculate the number of insertions needed for the current element
            if a[i] > i + 1:
                insertions += a[i] - (i + 1)
        
        results.append(insertions)
    
    return results

# Read input
t = int(input())
test_cases = []

for _ in range(t):
    n = int(input())
    a = list(map(int, input().split()))
    test_cases.append((n, a))

# Get results
results = min_operations_to_satisfy_condition(t, test_cases)

# Print results
for result in results:
    print(result)
