def solve():
    n, a, b = map(int, input().split())

    if abs(a - b) > 1 or a + b > n - 2:
        print(-1)
        return

    p = [0] * n
    
    if a > b:
        
        p[0] = n
        cur_max = n - 1
        cur_min = 1
        
        for i in range(1, n, 2):
            if a > 0:
                p[i] = cur_min
                cur_min += 1
                a -= 1
            else:
                p[i] = cur_max
                cur_max -= 1
        
        for i in range(2, n, 2):
            p[i] = cur_max
            cur_max -= 1

    elif b > a:
        p[0] = 1
        cur_min = 2
        cur_max = n
        
        for i in range(1, n, 2):
            if b > 0:
                p[i] = cur_max
                cur_max -= 1
                b -= 1
            else:
                p[i] = cur_min
                cur_min += 1
        
        for i in range(2, n, 2):
            p[i] = cur_min
            cur_min += 1
            
    else:
        
        p[0] = n
        cur_max = n - 1
        cur_min = 1
        
        for i in range(1, n, 2):
            if a > 0:
                p[i] = cur_min
                cur_min += 1
                a -= 1
            else:
                p[i] = cur_max
                cur_max -= 1
        
        for i in range(2, n, 2):
            p[i] = cur_max
            cur_max -= 1

    print(*p)

t = int(input())
for _ in range(t):
    solve()
