def find_permutation(t, test_cases):
    for _ in range(t):
        n, a, b = test_cases[_]
        
        if a + b + 1 > n or a * b == 0 and a + b > 0:
            print(-1)
            continue
        
        result = []
        for i in range(n - b, n):
            result.append(i + 1)
        
        for i in range(a):
            result.insert(i * 2, n - b - i)
        
        for i in range(n - a - b):
            result.insert(i * 2, i + 1)
        
        print(" ".join(map(str, result)))

# Input reading
t = int(input())
test_cases = []
for _ in range(t):
    n, a, b = map(int, input().split())
    test_cases.append((n, a, b))

# Find and output the permutations
find_permutation(t, test_cases)
